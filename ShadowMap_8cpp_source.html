17 bool ShadowMap::Initialize(ID3D12Device* device, uint32 width, uint32 height, uint32 arraySize, bool isCube)
18 {
19 m_width = width;
20 m_height = height;
21 m_arraySize = arraySize;
22 m_dsvDescriptorSize = device->GetDescriptorHandleIncrementSize(D3D12_DESCRIPTOR_HEAP_TYPE_DSV);
23
24 // 1. DSVヒープの作成
25 D3D12_DESCRIPTOR_HEAP_DESC dsvHeapDesc = {};
26 dsvHeapDesc.NumDescriptors = arraySize;
27 dsvHeapDesc.Type = D3D12_DESCRIPTOR_HEAP_TYPE_DSV;
28 dsvHeapDesc.Flags = D3D12_DESCRIPTOR_HEAP_FLAG_NONE;
29 if (FAILED(device->CreateDescriptorHeap(&dsvHeapDesc, IID_PPV_ARGS(&m_dsvHeap)))) return false;
30
31 // 2. SRVヒープの作成
32 D3D12_DESCRIPTOR_HEAP_DESC srvHeapDesc = {};
33 srvHeapDesc.NumDescriptors = 1;
34 srvHeapDesc.Type = D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV;
35 srvHeapDesc.Flags = D3D12_DESCRIPTOR_HEAP_FLAG_SHADER_VISIBLE; // シェーダーからアクセス可能にする
36 if (FAILED(device->CreateDescriptorHeap(&srvHeapDesc, IID_PPV_ARGS(&m_srvHeap)))) return false;
37
38 // 3. テクスチャリソースの作成
39 D3D12_RESOURCE_DESC texDesc = {};
40 texDesc.Dimension = D3D12_RESOURCE_DIMENSION_TEXTURE2D;
41 texDesc.Alignment = 0;
42 texDesc.Width = width;
43 texDesc.Height = height;
44 texDesc.DepthOrArraySize = arraySize;
45 texDesc.MipLevels = 1;
46 texDesc.Format = DXGI_FORMAT_R32_TYPELESS;
47 texDesc.SampleDesc.Count = 1;
48 texDesc.SampleDesc.Quality = 0;
49 texDesc.Layout = D3D12_TEXTURE_LAYOUT_UNKNOWN;
50 texDesc.Flags = D3D12_RESOURCE_FLAG_ALLOW_DEPTH_STENCIL;
51
52 D3D12_CLEAR_VALUE optClear = {};
53 optClear.Format = DXGI_FORMAT_D32_FLOAT;
54 optClear.DepthStencil.Depth = 1.0f;
55 optClear.DepthStencil.Stencil = 0;
56
57 D3D12_HEAP_PROPERTIES heapProps = {};
58 heapProps.Type = D3D12_HEAP_TYPE_DEFAULT;
59
60 if (FAILED(device->CreateCommittedResource(
61 &heapProps, D3D12_HEAP_FLAG_NONE,
62 &texDesc, D3D12_RESOURCE_STATE_PIXEL_SHADER_RESOURCE,
63 &optClear, IID_PPV_ARGS(&m_resource))))
64 {
65 return false;
66 }
67
68 // 4. DSV (書き込み用) の作成
69 D3D12_CPU_DESCRIPTOR_HANDLE dsvHandle = m_dsvHeap->GetCPUDescriptorHandleForHeapStart();
70 for (uint32 i = 0; i < arraySize; ++i)
71 {
72 D3D12_DEPTH_STENCIL_VIEW_DESC dsvDesc = {};
73 dsvDesc.Flags = D3D12_DSV_FLAG_NONE;
74 dsvDesc.Format = DXGI_FORMAT_D32_FLOAT;
75
76 if (arraySize > 1)
77 {
78 dsvDesc.ViewDimension = D3D12_DSV_DIMENSION_TEXTURE2DARRAY;
79 dsvDesc.Texture2DArray.FirstArraySlice = i;
80 dsvDesc.Texture2DArray.ArraySize = 1;
81 dsvDesc.Texture2DArray.MipSlice = 0;
82 }
83 else
84 {
85 dsvDesc.ViewDimension = D3D12_DSV_DIMENSION_TEXTURE2D;
86 dsvDesc.Texture2D.MipSlice = 0;
87 }
88
89 device->CreateDepthStencilView(m_resource.Get(), &dsvDesc, dsvHandle);
90 dsvHandle.ptr += m_dsvDescriptorSize;
91 }
92
93 // 5. SRV (読み込み用) の作成
94 D3D12_SHADER_RESOURCE_VIEW_DESC srvDesc = {};
95 srvDesc.Shader4ComponentMapping = D3D12_DEFAULT_SHADER_4_COMPONENT_MAPPING;
96 srvDesc.Format = DXGI_FORMAT_R32_FLOAT;
97
98 if (isCube && arraySize == 6) // Point Light
99 {
100 srvDesc.ViewDimension = D3D12_SRV_DIMENSION_TEXTURECUBE;
101 srvDesc.TextureCube.MostDetailedMip = 0;
102 srvDesc.TextureCube.MipLevels = 1;
103 srvDesc.TextureCube.ResourceMinLODClamp = 0.0f;
104 }
105 else if (arraySize > 1) // Spot Light
106 {
107 srvDesc.ViewDimension = D3D12_SRV_DIMENSION_TEXTURE2DARRAY;
108 srvDesc.Texture2DArray.FirstArraySlice = 0;
109 srvDesc.Texture2DArray.ArraySize = arraySize;
110 srvDesc.Texture2DArray.MipLevels = 1;
111 srvDesc.Texture2DArray.MostDetailedMip = 0;
112 }
113 else // Dir Light
114 {
115 srvDesc.ViewDimension = D3D12_SRV_DIMENSION_TEXTURE2D;
116 srvDesc.Texture2D.MipLevels = 1;
117 }
118
119 device->CreateShaderResourceView(m_resource.Get(), &srvDesc, m_srvHeap->GetCPUDescriptorHandleForHeapStart());
120
121 return true;
122 }
