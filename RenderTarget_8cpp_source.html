85 bool RenderTarget::Resize(ID3D12Device* device, uint32 newWidth, uint32 newHeight)
86 {
87 if (newWidth == 0 || newHeight == 0) return false;
88
89 // 最小サイズ保証
90 newWidth = std::max(newWidth, 1u);
91 newHeight = std::max(newHeight, 1u);
92
93 if (width == newWidth && height == newHeight && resource) return true;
94
95 width = newWidth;
96 height = newHeight;
97
98 // 既存のリソースがあれば解放
99 resource.Reset();
100 depthResource.Reset();
101
102 // リソース記述
103 D3D12_RESOURCE_DESC resDesc = {};
104 resDesc.Dimension = D3D12_RESOURCE_DIMENSION_TEXTURE2D;
105 resDesc.Alignment = 0;
106 resDesc.Width = width;
107 resDesc.Height = height;
108 resDesc.DepthOrArraySize = 1;
109 resDesc.MipLevels = 1;
110 resDesc.Format = format;
111 resDesc.SampleDesc.Count = 1;
112 resDesc.SampleDesc.Quality = 0;
113 resDesc.Layout = D3D12_TEXTURE_LAYOUT_UNKNOWN;
114 resDesc.Flags = D3D12_RESOURCE_FLAG_ALLOW_RENDER_TARGET;
115
116 // クリア値の設定
117 D3D12_CLEAR_VALUE optClear;
118 optClear.Format = format;
119 memcpy(optClear.Color, clearColor, sizeof(float) * 4);
120
121 // デフォルトヒープ（VRAM）に作成
122 D3D12_HEAP_PROPERTIES heapProps = {};
123 heapProps.Type = D3D12_HEAP_TYPE_DEFAULT;
124 heapProps.CPUPageProperty = D3D12_CPU_PAGE_PROPERTY_UNKNOWN;
125 heapProps.MemoryPoolPreference = D3D12_MEMORY_POOL_UNKNOWN;
126 heapProps.CreationNodeMask = 1;
127 heapProps.VisibleNodeMask = 1;
128
129 // 初期状態は PIXEL_SHADER_RESOURCE にしておく（ImGuiが表示するため）
130 currentState = D3D12_RESOURCE_STATE_PIXEL_SHADER_RESOURCE;
131
132 HRESULT hr = device->CreateCommittedResource(
133 &heapProps,
134 D3D12_HEAP_FLAG_NONE,
135 &resDesc,
136 currentState,
137 &optClear,
138 IID_PPV_ARGS(&resource)
139 );
140
141 if (FAILED(hr))
142 {
143 SPAN_ERROR("RenderTarget: Failed to create texture resource!");
144 return false;
145 }
146
147 CreateViews(device);
148 CreateDepthBuffer(device);
149
150 return true;
151 }
