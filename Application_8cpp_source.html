14 {
15 // コンソールのxボタンが押された場合

139 void Application::Run()
140 {
141 if (!isRunning) return;
142
143 // ユーザー定義の開始処理
144 OnStart();
145
146 // Main Loop
147 // ------------------------------------------------------------
148 while (isRunning)
149 {
150 if (!window.PollEvents())
151 {
152 isRunning = false;
153 break;
154 }
155
156 if (window.GetWidth() == 0 || window.GetHeight() == 0)
157 {
158 Sleep(10);
159 continue;
160 }
161
162 // 1. SceneBuffer Resize Check
163 // ------------------------------------------------------------
164 if (auto scenePanel = GuiManager::GetPanel<SceneViewPanel>())
165 {
166 Vector2 viewSize = scenePanel->GetTargetResolution();
167
168 // サイズが有効かつ、現在と異なる場合のみリサイズ
169 if (viewSize.x > 1.0f && viewSize.y > 1.0f)
170 {
171 uint32 targetW = (uint32)viewSize.x;
172 uint32 targetH = (uint32)viewSize.y;
173
174 // サイズが異なる場合のみ再構築
175 if (sceneBuffer.GetWidth() != targetW || sceneBuffer.GetHeight() != targetH)
176 {
177 graphicsContext.WaitForGpu();
178 sceneBuffer.Resize(renderer.GetDevice(), targetW, targetH);
179
180 // 更新されたサイズを保持
181 m_sceneViewWidth = targetW;
182 m_sceneViewHeight = targetH;
183 }
184 }
185 }
186
187 // 2. Scene Rendering
188 // ------------------------------------------------------------
189 // コマンドリスト取得
190 ID3D12GraphicsCommandList* cmd = renderer.BeginFrame();
191
192 graphicsContext.SetRenderTargetToBackBuffer(cmd);
193 GuiManager::BeginFrame();
194
195 // ドッキングスペースの作成
196 ImGuiID dockspace_id = ImGui::GetID("MyDockSpace");
197 ImGui::DockSpaceOverViewport(dockspace_id, ImGui::GetMainViewport());
198
199 // --- A. シーン描画 (Render to Texture) ---
200 {
201 // バリア: SRV -> RTV
202 sceneBuffer.TransitionToRenderTarget(cmd);
203
204 // レンダーターゲット設定
205 D3D12_CPU_DESCRIPTOR_HANDLE rtv = sceneBuffer.GetRTV();
206 D3D12_CPU_DESCRIPTOR_HANDLE dsv = sceneBuffer.GetDSV();
207 cmd->OMSetRenderTargets(1, &rtv, FALSE, &dsv);
208
209 sceneBuffer.Clear(cmd);
210
211 // クリア & ビューポート
212 D3D12_VIEWPORT viewport = { 0.0f, 0.0f, (float)sceneBuffer.GetWidth(), (float)sceneBuffer.GetHeight(), 0.0f, 1.0f };
213 D3D12_RECT scissor = { 0, 0, (LONG)sceneBuffer.GetWidth(), (LONG)sceneBuffer.GetHeight() };
214 cmd->RSSetViewports(1, &viewport);
215 cmd->RSSetScissorRects(1, &scissor);
216
217 // Logic Update & ECS Draw
218 Time::Update(); // Time update
219 Input::Update(); // Input update
220 GetWorld().UpdateSystems(); // Systems update
221 OnUpdate(); // User update
222
223 GetWorld().ForEach<Camera, Transform>([&](Entity, Camera&, Transform& t)
224 {
225 renderer.SetCameraPosition(t.Position);
226 });
227
228 // グリッド描画
229 renderer.RenderGrid(cmd);
230
231 // バリア: RTV -> SRV
232 sceneBuffer.TransitionToShaderResource(cmd);
233 }
234
235 // 3. Editor UI Rendering
236 // ------------------------------------------------------------
237 graphicsContext.SetRenderTargetToBackBuffer(cmd);
238
239 // シーンビューへのテクスチャセット
240 if (auto scenePanel = GuiManager::GetPanel<SceneViewPanel>())
241 {
242 D3D12_CPU_DESCRIPTOR_HANDLE srvHandle = sceneBuffer.GetSRV();
243 if (srvHandle.ptr != 0)
244 {
245 D3D12_GPU_DESCRIPTOR_HANDLE imGuiTexture = GuiManager::RegisterTexture(srvHandle);
246 scenePanel->SetTexture(imGuiTexture);
247 }
248 }
249
250 // 描画終了
251 GuiManager::EndFrame(cmd);
252
253 renderer.EndFrame();
254 Input::EndFrame();
255 }
256
257 OnShutdown();
258 }
