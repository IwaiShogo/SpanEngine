31 void DirectoryWatcher::SetDirectory(const std::filesystem::path& newDirectory)
32 {
33 // 1. スレッドの停止要求を出す
34 m_IsRunning = false;
35
36 // 2. 監視スレッドの終了を待つ
37 if (m_WatchThread.joinable())
38 {
39 m_WatchThread.join();
40 }
41
42 // 3. スレッドが完全に停止した状態で、安全にデータを更新する
43 {
44 std::lock_guard<std::mutex> lock(m_Mutex);
45
46 m_CurrentDirectory = newDirectory;
47 m_FileCache.clear();
48
49 // 初回状態をキャッシュ
50 if (std::filesystem::exists(m_CurrentDirectory))
51 {
52 for (const auto& entry : std::filesystem::directory_iterator(m_CurrentDirectory))
53 {
54 m_FileCache[entry.path().string()] = std::filesystem::last_write_time(entry);
55 }
56 }
57 }
58
59 // スレッド再開
60 m_IsRunning = true;
61 m_WatchThread = std::thread(&DirectoryWatcher::WatchLoop, this);
62 }

64 void DirectoryWatcher::WatchLoop()
65 {
66 while (m_IsRunning)
67 {
68 // 500ミリ秒毎にポーリング
69 std::this_thread::sleep_for(std::chrono::milliseconds(500));
70
71 std::lock_guard<std::mutex> lock(m_Mutex);
72 if (!std::filesystem::exists(m_CurrentDirectory)) continue;
73
74 bool changed = false;
75 std::unordered_map<std::string, std::filesystem::file_time_type> currentFiles;
76
77 // 現在のファイル状態を取得
78 for (const auto& entry : std::filesystem::directory_iterator(m_CurrentDirectory))
79 {
80 currentFiles[entry.path().string()] = std::filesystem::last_write_time(entry);
81 }
82
83 // 1. 追加・変更の検知
84 for (const auto& [path, time] : currentFiles)
85 {
86 auto it = m_FileCache.find(path);
87 if (it == m_FileCache.end() || it->second != time)
88 {
89 changed = true;
90 break;
91 }
92 }
93
94 // 2. 削除の検知
95 if (!changed)
96 {
97 for (const auto& [path, time] : m_FileCache)
98 {
99 if (currentFiles.find(path) == currentFiles.end())
100 {
101 changed = true;
102 break;
103 }
104 }
105 }
106
107 // 変更があればコールバックを発火してキャッシュ更新
108 if (changed)
109 {
110 m_FileCache = currentFiles;
111 if (m_OnAction && m_IsRunning)
112 {
113 m_OnAction();
114 }
115 }
116 }
117 }
