29 void EnvironmentPanel::OnImGuiRender()
30 {
31 // ウィンドウの初期サイズを設定
32 ImGui::SetNextWindowSize(ImVec2(400, 500), ImGuiCond_FirstUseEver);
33
34 if (ImGui::Begin(title.c_str(), &isOpen))
35 {
36 // 現在のアクティブシーンから環境設定を取得
37 auto& env = Application::Get().GetActiveScene().Environment;
38
39 ImGui::PushStyleVar(ImGuiStyleVar_FramePadding, ImVec2(4, 4));
40
41 // --- Skybox Settings ---
42 if (ImGui::CollapsingHeader("Skybox Settings", ImGuiTreeNodeFlags_DefaultOpen))
43 {
44 ImGui::Indent(10.0f);
45
46 const char* modeStrings[] = { "Procedural (3-Color Gradient)", "HDRI (High Dynamic Range Image)" };
47 int currentMode = static_cast<int>(env.Mode);
48 if (ImGui::Combo("Sky Mode", &currentMode, modeStrings, IM_ARRAYSIZE(modeStrings)))
49 {
50 env.Mode = static_cast<SkyboxMode>(currentMode);
51 }
52
53 ImGui::Spacing();
54
55 if (env.Mode == SkyboxMode::Procedural)
56 {
57 ImGuiColorEditFlags hdrFlags = ImGuiColorEditFlags_Float | ImGuiColorEditFlags_HDR;
58 ImGui::ColorEdit3("Top Color", &env.SkyTopColor.x, hdrFlags);
59 ImGui::ColorEdit3("Horizon Color", &env.SkyHorizonColor.x, hdrFlags);
60 ImGui::ColorEdit3("Bottom Color", &env.SkyBottomColor.x, hdrFlags);
61
62 if (ImGui::Button("Reset Colors"))
63 {
64 env.SkyTopColor = { 0.35f, 0.5f, 0.7f };
65 env.SkyHorizonColor = { 0.7f, 0.75f, 0.8f };
66 env.SkyBottomColor = { 0.2f, 0.2f, 0.2f };
67 }
68 }
69 else if (env.Mode == SkyboxMode::HDRI)
70 {
71 char pathBuffer[256];
72 strcpy_s(pathBuffer, env.HDRIPath.c_str());
73
74 ImGui::Text("HDRI Asset Path");
75 ImGui::SetNextItemWidth(ImGui::GetContentRegionAvail().x - 40.0f);
76 if (ImGui::InputText("##HDRIPath", pathBuffer, sizeof(pathBuffer)))
77 {
78 env.HDRIPath = pathBuffer;
79 }
80
81 ImGui::SameLine();
82 if (ImGui::Button("X", ImVec2(30, 0))) { env.HDRIPath = ""; }
83
84 // D&Dサポート
85 if (ImGui::BeginDragDropTarget())
86 {
87 if (const ImGuiPayload* payload = ImGui::AcceptDragDropPayload("CONTENT_BROWSER_ITEM"))
88 {
89 const wchar_t* path = (const wchar_t*)payload->Data;
90 std::wstring ws(path);
91 env.HDRIPath = std::string(ws.begin(), ws.end());
92 }
93 ImGui::EndDragDropTarget();
94 }
95 ImGui::TextColored(ImVec4(0.5f, 0.5f, 0.5f, 1.0f), "Tip: Drop .hdr file here");
96 }
97
98 ImGui::Unindent(10.0f);
99 }
100
101 // --- Global Lighting Settings ---
102 if (ImGui::CollapsingHeader("Global Lighting", ImGuiTreeNodeFlags_DefaultOpen))
103 {
104 ImGui::Indent(10.0f);
105 ImGui::SliderFloat("Camera Exposure", &env.Exposure, 0.1f, 10.0f, "%.2f");
106 ImGui::SliderFloat("Ambient Intensity", &env.AmbientIntensity, 0.0f, 5.0f, "%.2f");
107 ImGui::SliderFloat("Reflection Intensity", &env.EnvReflectionIntensity, 0.0f, 10.0f, "%.2f");
108 ImGui::Unindent(10.0f);
109 }
110
111 ImGui::PopStyleVar();
112 }
113 ImGui::End();
114 }
