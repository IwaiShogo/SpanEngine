136 void SSAOPass::GenerateSampleKernel()
137 {
138 std::uniform_real_distribution<float> randomFloats(0.0f, 1.0f);
139 std::default_random_engine generator;
140
141 for (int i = 0; i < 64; ++i)
142 {
143 // X, Y„ÅØ -1.0 ~ 1.0, Z„ÅØ 0.0 ~ 1.0 (ÂçäÁêÉ)
144 Vector3 sample(
145 randomFloats(generator) * 2.0f - 1.0f,
146 randomFloats(generator) * 2.0f - 1.0f,
147 randomFloats(generator)
148 );
149 sample = Vector3::Normalize(sample);
150 sample = sample * randomFloats(generator);

161 void SSAOPass::GenerateNoiseTexture(ID3D12Device* device, ID3D12CommandQueue* commandQueue)
162 {
163 std::uniform_real_distribution<float> randomFloats(0.0f, 1.0f);
164 std::default_random_engine generator;
165
166 std::vector<uint32> noiseData(16); // 4x4
167 for (int i = 0; i < 16; ++i)
168 {
169 // Z(B) „ÅØ 0„ÄÅXY(RG) „Å´„É©„É≥„ÉÄ„É†„Å™ÂõûËª¢ÊñπÂêë„ÇíÊåÅ„Åü„Åõ„Çã
170 float x = randomFloats(generator) * 2.0f - 1.0f;
171 float y = randomFloats(generator) * 2.0f - 1.0f;
172 Vector3 noise(x, y, 0.0f);
173 noise = Vector3::Normalize(noise);
174
175 // -1~1 „Çí 0~255 „ÅÆ RGBA „Å´Â§âÊèõ
176 uint8 r = (uint8)((noise.x * 0.5f + 0.5f) * 255.0f);
177 uint8 g = (uint8)((noise.y * 0.5f + 0.5f) * 255.0f);
178 uint8 b = 0;
