23 bool RenderPassManager::Initialize(GraphicsContext* context)
24 {
25 if (!context) return false;
26 auto device = context->GetDevice();
27 auto queue = context->GetCommandQueue();
28 uint32 w = context->GetViewportWidth();
29 uint32 h = context->GetViewportHeight();

31 m_gridPass = std::make_unique<GridPass>();

39
40 m_spotShadowPass = std::make_unique<ShadowPass>();
41 if (!m_spotShadowPass->Initialize(device, 1024, 1024, 4)) return false;
42
43 m_pointShadowPass = std::make_unique<ShadowPass>();
44 if (!m_pointShadowPass->Initialize(device, 1024, 1024, 6, true)) return false;
45
46 m_depthNormalPass = std::make_unique<DepthNormalPass>();
47 if (!m_depthNormalPass->Initialize(device, w, h)) return false;
48
49 m_ssaoPass = std::make_unique<SSAOPass>();
50 if (!m_ssaoPass->Initialize(device, queue, w, h)) return false;
51
52 m_ssaoBlurPass = std::make_unique<SSAOBlurPass>();
