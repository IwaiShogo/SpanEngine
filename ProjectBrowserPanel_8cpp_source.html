159 void ProjectBrowserPanel::DrawDirectoryNode(const std::filesystem::path& path)
160 {
161 // ディレクトリ名を取得
162 std::string filename = path.filename().string();
163 if (filename.empty()) filename = path.string();
164
165 // ツリーノードのフラグ
166 ImGuiTreeNodeFlags flags = ImGuiTreeNodeFlags_OpenOnArrow | ImGuiTreeNodeFlags_OpenOnDoubleClick | ImGuiSelectableFlags_SpanAvailWidth;
167
168 // カレントディレクトリなら選択状態にする
169 if (path == m_CurrentDirectory)
170 {
171 flags |= ImGuiTreeNodeFlags_Selected;
172 }
173
174 // サブディレクトリがあるか確認 (Leaf判定)
175 bool isLeaf = true;
176 try
177 {
178 for (const auto& entry : std::filesystem::directory_iterator(path))
179 {
180 if (entry.is_directory())
181 {
182 isLeaf = false;
183 break;
184 }
185 }
186 }
187 catch (...) {}
188
189 if (isLeaf)
190 {
191 flags |= ImGuiTreeNodeFlags_Leaf;
192 }
193
194 // ノード描画
195 bool opened = ImGui::TreeNodeEx(path.string().c_str(), flags, "%s %s", ICON_FA_FOLDER, filename.c_str());
196
197 // クリックでディレクトリ移動
198 if (ImGui::IsItemClicked())
199 {
200 m_CurrentDirectory = path;
201 }
202
203 // 再帰描画
204 if (opened)
205 {
206 if (!isLeaf)
207 {
208 try
209 {
210 for (const auto& entry : std::filesystem::directory_iterator(path))
211 {
212 if (entry.is_directory())
213 {
214 // 隠しフォルダスキップ
215 if (!m_ShowHiddenFiles && entry.path().filename().string()[0] == '.') continue;
216 DrawDirectoryNode(entry.path());
217 }
218 }
219 }
220 catch (...) {}
221 }
222 ImGui::TreePop();
223 }

226 void ProjectBrowserPanel::DrawContentArea()
