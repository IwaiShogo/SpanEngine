24 void AssetRegistry::Refresh(const std::filesystem::path& rootDirectory)
25 {
26 m_Assets.clear();
27 m_PathToHandle.clear();
28
29 SPAN_LOG("AssetRegistry: Scanning assets in %s...", rootDirectory.string().c_str());
30
31 if (!std::filesystem::exists(rootDirectory)) return;
32
33 for (const auto& entry : std::filesystem::recursive_directory_iterator(rootDirectory))
34 {
35 if (entry.is_regular_file())
36 {
37 const auto& path = entry.path();
38
39 // .meta ファイル自体は登録しない
40 if (path.extension() == ".meta") continue;
41
42 // 対応する .meta があるか確認して登録
43 RegisterAsset(path);
44 }
45 }
46
47 SPAN_LOG("AssetRegistry: Scan complete. %d assets indexed.", m_Assets.size());
48 }

50 void AssetRegistry::RegisterAsset(const std::filesystem::path& path)
51 {
52 // .meta が存在するか確認し、読み込む
53 if (std::filesystem::exists(path.string() + ".meta"))
54 {
55 AssetMetadata meta = AssetSerializer::LoadOrCreateMetadata(path);
56 if (meta.IsValid())
57 {
58 m_Assets[meta.Handle] = path;
59 m_PathToHandle[path.string()] = meta.Handle;
60 }
61 }
62 }

64 void AssetRegistry::UnregisterAsset(const std::filesystem::path& path)
65 {
66 auto it = m_PathToHandle.find(path.string());
67 if (it != m_PathToHandle.end())
68 {
69 m_Assets.erase(it->second);
70 m_PathToHandle.erase(it);
71 }
72 }
