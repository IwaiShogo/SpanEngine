40 /**
41 * @brief 新しいタグを追加します。
42 * @param tag 追加するタグ文字列
43 * @return true 追加成功 (または既に存在) / false 不正な文字列
44 */
45 bool AddTag(const std::string& tag)
46 {
47 if (tag.empty()) return false;
48 if (HasTag(tag)) return true;
49
50 m_Tags.push_back(tag);
51 return true;
52 }

54 /**
55 * @brief 指定したタグを削除します。("Untagged"は削除不可)
56 * @param tag 削除するタグ文字列
57 * @return true 削除成功 / false 削除失敗 (存在しない、または保護されたタグ)
58 */
59 bool RemoveTag(const std::string& tag)
60 {
61 if (tag == "Untagged") return false;
62
63 auto it = std::find(m_Tags.begin(), m_Tags.end(), tag);
64 if (it != m_Tags.end())
65 {
66 m_Tags.erase(it);
67 return true;
68 }
69 return false;
70 }
71
72 /**
73 * @brief タグが存在するかどうかを確認します。
74 * @param tag 確認するタグ文字列
75 * @return true 存在する / false 存在しない
76 */
77 bool HasTag(const std::string& tag) const
78 {
79 return std::find(m_Tags.begin(), m_Tags.end(), tag) != m_Tags.end();
80 }
81
82 private:
