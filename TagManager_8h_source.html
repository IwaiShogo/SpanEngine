21 class TagManager
22 {
23 public:
24 /**

26 * @return TagManager& インスタンス参照
27 */
28 static TagManager& Get()

34 /**
35 * @brief 登録されているすべてのタグリストを取得します。
36 * @return const std::vector<std::string>& タグのリスト
37 */
38 const std::vector<std::string>& GetAllTags() const { return m_Tags; }
39
40 /**
41 * @brief 文字列がタブとして有効化(英数字とアンダースコアのみか)判定します。
42 */
43 bool IsValidTagName(const std::string& tag) const
44 {
45 if (tag.empty()) return false;
46 return std::all_of(tag.begin(), tag.end(), [](unsigned char c) { return std::isalnum(c) || c == '_'; });
47 }
48
49 /**
50 * @brief システムが保護しているタグかどうか判定します(削除・変更不可)。
51 */
52 bool IsProtectedTag(const std::string& tag) const
53 {
54 return tag == "Untagged" || tag == "Player" || tag == "Enemy" || tag == "MainCamera";
55 }
56
57 /**
58 * @brief 新しいタグを追加します。
59 * @param tag 追加するタグ文字列
60 * @return true 追加成功 (または既に存在) / false 不正な文字列
61 */
62 bool AddTag(const std::string& tag)
63 {
64 if (!IsValidTagName(tag)) return false;
65 if (HasTag(tag)) return true;
66
67 m_Tags.push_back(tag);
68 SortTags();
69 return true;
70 }
71
72 /**
73 * @brief 指定したタグを削除します。(保護されたタグは削除不可)
74 * @param tag 削除するタグ文字列
75 * @return true 削除成功 / false 削除失敗 (存在しない、または保護されたタグ)
76 */
77 bool RemoveTag(const std::string& tag)
78 {
79 if (IsProtectedTag(tag)) return false;
80
81 auto it = std::find(m_Tags.begin(), m_Tags.end(), tag);
82 if (it != m_Tags.end())
83 {
84 m_Tags.erase(it);
85 return true;
86 }
87 return false;
88 }
89
90 /**
91 * @brief タグが存在するかどうかを確認します。
92 * @param tag 確認するタグ文字列
93 * @return true 存在する / false 存在しない
94 */
95 bool HasTag(const std::string& tag) const
96 {
97 return std::find(m_Tags.begin(), m_Tags.end(), tag) != m_Tags.end();
98 }
99
100 private:
101 TagManager()
102 {
103 // デフォルトで用意される基本タグ
104 m_Tags = { "Untagged", "Player", "Enemy", "MainCamera" };
105 }
106 ~TagManager() = default;
107
108 /**
109 * @brief "Untagged" を必ず先頭にし、残りをアルファベット順にソートします。
110 */
111 void SortTags()
112 {
113 if (m_Tags.empty()) return;
114
115 auto it = std::find(m_Tags.begin(), m_Tags.end(), "Untagged");
116 if (it != m_Tags.end() && it != m_Tags.begin())
117 {
118 std::swap(m_Tags[0], *it);
119 }
120
121 if (m_Tags.size() > 1)
122 {
123 std::sort(m_Tags.begin() + 1, m_Tags.end());
124 }
125 }
126
127 private:
128 std::vector<std::string> m_Tags;
129 };
