46 void LayerManager::SetCollision(uint8_t layerA, uint8_t layerB, bool canCollide)
47 {
48 if (layerA >= 32 || layerB >= 32) return;
49
50 if (canCollide)
51 {
52 m_CollisionMatrix[layerA] |= (1 << layerB);
53 m_CollisionMatrix[layerB] |= (1 << layerA);
54 }
55 else
56 {
57 m_CollisionMatrix[layerA] &= ~(1 << layerB);
58 m_CollisionMatrix[layerB] &= ~(1 << layerA);
59 }
60 }
