16 bool ComputeBuffer::Initialize(ID3D12Device* device, uint32 elementSize, uint32 elementCount, bool isUAV)
17 {
18 m_elementSize = elementSize;
19 m_elementCount = elementCount;
20 m_isUAV = isUAV;
21
22 uint64 bufferSize = (uint64)elementSize * elementCount;
23
24 D3D12_HEAP_PROPERTIES heapProps = {};
25 heapProps.Type = isUAV ? D3D12_HEAP_TYPE_DEFAULT : D3D12_HEAP_TYPE_UPLOAD;
26
27 D3D12_RESOURCE_DESC desc = {};
28 desc.Dimension = D3D12_RESOURCE_DIMENSION_BUFFER;
29 desc.Width = bufferSize;
30 desc.Height = 1;
31 desc.DepthOrArraySize = 1;
32 desc.MipLevels = 1;
33 desc.Format = DXGI_FORMAT_UNKNOWN;
34 desc.SampleDesc.Count = 1;
35 desc.Layout = D3D12_TEXTURE_LAYOUT_ROW_MAJOR;
36 desc.Flags = isUAV ? D3D12_RESOURCE_FLAG_ALLOW_UNORDERED_ACCESS : D3D12_RESOURCE_FLAG_NONE;
37
38 if (FAILED(device->CreateCommittedResource(
39 &heapProps, D3D12_HEAP_FLAG_NONE, &desc,
40 isUAV ? D3D12_RESOURCE_STATE_COMMON : D3D12_RESOURCE_STATE_GENERIC_READ,
41 nullptr, IID_PPV_ARGS(&m_resource))))
42 {
43 return false;
44 }
45
46 // Upload Heapの場合はメモリをマップしっぱなしにする
47 if (!isUAV)
48 {
49 D3D12_RANGE readRange = { 0, 0 };
50 m_resource->Map(0, &readRange, reinterpret_cast<void**>(&m_mappedData));
51 }
52
53 // SRV の作成
54 D3D12_DESCRIPTOR_HEAP_DESC srvHeapDesc = {};
55 srvHeapDesc.NumDescriptors = 1;
56 srvHeapDesc.Type = D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV;
57 srvHeapDesc.Flags = D3D12_DESCRIPTOR_HEAP_FLAG_SHADER_VISIBLE;
58 device->CreateDescriptorHeap(&srvHeapDesc, IID_PPV_ARGS(&m_srvHeap));
59
60 D3D12_SHADER_RESOURCE_VIEW_DESC srvDesc = {};
61 srvDesc.Shader4ComponentMapping = D3D12_DEFAULT_SHADER_4_COMPONENT_MAPPING;
62 srvDesc.Format = DXGI_FORMAT_UNKNOWN;
63 srvDesc.ViewDimension = D3D12_SRV_DIMENSION_BUFFER;
64 srvDesc.Buffer.FirstElement = 0;
65 srvDesc.Buffer.NumElements = elementCount;
66 srvDesc.Buffer.StructureByteStride = elementSize;
67 srvDesc.Buffer.Flags = D3D12_BUFFER_SRV_FLAG_NONE;
68 device->CreateShaderResourceView(m_resource.Get(), &srvDesc, m_srvHeap->GetCPUDescriptorHandleForHeapStart());
69
70 // UAV の作成
71 if (isUAV)
72 {
73 D3D12_DESCRIPTOR_HEAP_DESC uavHeapDesc = {};
74 uavHeapDesc.NumDescriptors = 1;
75 uavHeapDesc.Type = D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV;
76 uavHeapDesc.Flags = D3D12_DESCRIPTOR_HEAP_FLAG_SHADER_VISIBLE;
77 device->CreateDescriptorHeap(&uavHeapDesc, IID_PPV_ARGS(&m_uavHeap));
78
79 D3D12_UNORDERED_ACCESS_VIEW_DESC uavDesc = {};
80 uavDesc.Format = DXGI_FORMAT_UNKNOWN;
81 uavDesc.ViewDimension = D3D12_UAV_DIMENSION_BUFFER;
82 uavDesc.Buffer.FirstElement = 0;
83 uavDesc.Buffer.NumElements = elementCount;
84 uavDesc.Buffer.StructureByteStride = elementSize;
85 uavDesc.Buffer.CounterOffsetInBytes = 0;
86 uavDesc.Buffer.Flags = D3D12_BUFFER_UAV_FLAG_NONE;
87 device->CreateUnorderedAccessView(m_resource.Get(), nullptr, &uavDesc, m_uavHeap->GetCPUDescriptorHandleForHeapStart());
88 }
89
90 return true;
91 }

93 void ComputeBuffer::Shutdown()
94 {
95 if (!m_isUAV && m_resource && m_mappedData)
96 {
97 m_resource->Unmap(0, nullptr);
98 m_mappedData = nullptr;
99 }
100
101 // ヒープ
102 m_srvHeap.Reset();
103 m_uavHeap.Reset();
104
105 m_resource.Reset();
106 }
