21 bool Renderer::Initialize(GraphicsContext* inContext)
22 {
23 this->context = inContext;
24 if (!context) return false;
25 auto device = context->GetDevice();
26
27 if (FAILED(device->CreateFence(0, D3D12_FENCE_FLAG_NONE, IID_PPV_ARGS(&m_waitFence)))) return false;
28 m_waitEvent = CreateEvent(nullptr, FALSE, FALSE, nullptr);
29 m_waitFenceValue = 1;
30
31 if (!CreateRootSignature()) return false;
32
33 vs = new Shader(); if (!vs->Load(L"Basic.hlsl", ShaderType::Vertex, "VSMain")) return false;
34 ps = new Shader(); if (!ps->Load(L"Basic.hlsl", ShaderType::Pixel, "PSMain")) return false;
35
36 if (!CreatePipelineState()) return false;
37 if (!CreateConstantBuffer()) return false;
38
39 // フレーム用の巨大な Descriptor Heap を作成
40 D3D12_DESCRIPTOR_HEAP_DESC heapDesc = {};
41 heapDesc.NumDescriptors = 4096;
42 heapDesc.Type = D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV;
43 heapDesc.Flags = D3D12_DESCRIPTOR_HEAP_FLAG_SHADER_VISIBLE;
44 if (FAILED(device->CreateDescriptorHeap(&heapDesc, IID_PPV_ARGS(&m_frameSrvHeap)))) return false;
45 m_srvDescriptorSize = device->GetDescriptorHandleIncrementSize(D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV);
46
47 // --- Passes Initialization ---
48 m_gridPass = std::make_unique<GridPass>();
49 if (!m_gridPass->Initialize(device)) return false;
50
51 m_skyboxPass = std::make_unique<SkyboxPass>();
52 if (!m_skyboxPass->Initialize(device)) return false;
53
54 m_dirShadowPass = std::make_unique<ShadowPass>();
55 if (!m_dirShadowPass->Initialize(device, 4096, 4096, 1)) return false;
56
57 m_spotShadowPass = std::make_unique<ShadowPass>();
58 if (!m_spotShadowPass->Initialize(device, 1024, 1024, 4)) return false;
59
60 m_pointShadowPass = std::make_unique<ShadowPass>();
61 if (!m_pointShadowPass->Initialize(device, 1024, 1024, 6, true)) return false;
62
63 m_LightBuffer = new ConstantBuffer<GlobalLightData>();
64 if (!m_LightBuffer->Initialize(device)) return false;
65
66 SPAN_LOG("Renderer Initialized Successfully with Render Passes!");
67 return true;
68 }

141 void Renderer::DrawMesh(Mesh* mesh, Material* material, const Matrix4x4& worldMatrix)
142 {
143 if (!mesh || !material || !commandList) return;
144 auto device = context->GetDevice();
145
146 Matrix4x4 mvp = worldMatrix * viewMatrix * projectionMatrix;
147 TransformData data;
148 data.MVP.FromXM(XMMatrixTranspose(mvp.ToXM()));
149 data.World.FromXM(XMMatrixTranspose(worldMatrix.ToXM()));
150
151 D3D12_GPU_VIRTUAL_ADDRESS cbAddr = AllocateCBV(&data, sizeof(TransformData));
152 if (cbAddr == 0) return;
153
154 material->Update();
155 commandList->SetPipelineState(material->GetBlendMode() == BlendMode::Transparent ? pipelineStateTransparent.Get() : pipelineState.Get());
156 commandList->SetGraphicsRootSignature(rootSignature.Get());
157
158 commandList->SetGraphicsRootConstantBufferView(0, cbAddr);
159 commandList->SetGraphicsRootConstantBufferView(1, material->GetGPUVirtualAddress());
160
161 if (m_LightBuffer) commandList->SetGraphicsRootConstantBufferView(15, m_LightBuffer->GetGPUVirtualAddress());
162
163 // PBR Textures (t0 ~ t5)
164 Texture* textures[6] = { material->GetAlbedoMap(), material->GetNormalMap(), material->GetMetallicMap(), material->GetRoughnessMap(), material->GetAOMap(), material->GetEmissiveMap() };
165 for (int i = 0; i < 6; i++) {
166 BindTexture(commandList, textures[i], 2 + i, D3D12_SRV_DIMENSION_TEXTURE2D);
167 }
168
169 // Shadows (t6 ~ t8)
170 BindShadowMap(commandList, m_dirShadowPass ? m_dirShadowPass->GetShadowMap() : nullptr, 8, D3D12_SRV_DIMENSION_TEXTURE2D);
171 BindShadowMap(commandList, m_spotShadowPass ? m_spotShadowPass->GetShadowMap() : nullptr, 9, D3D12_SRV_DIMENSION_TEXTURE2DARRAY);
172 BindShadowMap(commandList, m_pointShadowPass ? m_pointShadowPass->GetShadowMap() : nullptr, 10, D3D12_SRV_DIMENSION_TEXTURECUBE);
173
174 // IBL Textures (t9 ~ t11)
175 BindTexture(commandList, m_irradianceMap.get(), 11, D3D12_SRV_DIMENSION_TEXTURECUBE);
176 BindTexture(commandList, m_prefilterMap.get(), 12, D3D12_SRV_DIMENSION_TEXTURECUBE);
177 BindTexture(commandList, m_brdfLUT.get(), 13, D3D12_SRV_DIMENSION_TEXTURE2D);
178 BindTexture(commandList, m_opaqueCaptureTex.get(), 14, D3D12_SRV_DIMENSION_TEXTURE2D);
179
180 mesh->Draw(commandList);
181 }

219 bool Renderer::LoadEnvironmentMap(const std::string& filepath)
220 {
221 if (filepath.empty() || m_currentLoadedHDRI == filepath) return true;
222
223 auto device = context->GetDevice();
224 auto queue = context->GetCommandQueue();
225
226 // 1. パノラマHDRI画像を読み込む (.hdr)
227 Texture panoramaTex;
228 if (!panoramaTex.Initialize(device, queue, filepath))
229 {
230 SPAN_ERROR("Failed to load HDRI: %s", filepath.c_str());
231 return false;
232 }
233
234 // 2. 出力先の空の Cubemap を作成
235 m_envCubemap = std::make_unique<Texture>();
236 if (!m_envCubemap->InitializeAsCubemap(device, 1024)) return false;
237
238 // 3. Builderの初期化
239 IBLBuilder builder;
240 if (!builder.Initialize(device)) return false;
241
242 // 4. Compute Shader用の1回限りのコマンドリストを作成
243 ComPtr<ID3D12CommandAllocator> cmdAlloc;
244 ComPtr<ID3D12GraphicsCommandList> cmdList;
245 device->CreateCommandAllocator(D3D12_COMMAND_LIST_TYPE_DIRECT, IID_PPV_ARGS(&cmdAlloc));
246 device->CreateCommandList(0, D3D12_COMMAND_LIST_TYPE_DIRECT, cmdAlloc.Get(), nullptr, IID_PPV_ARGS(&cmdList));
247
248 // 5. 変換処理
249 builder.GenerateCubemapFromPanorama(device, cmdList.Get(), panoramaTex.GetCPUDescriptorHandle(), m_envCubemap.get(), 1024);
250
251 // m_envCubemap のステートを UAV から SRV に遷移
252 D3D12_RESOURCE_BARRIER envBarrier = {};
253 envBarrier.Type = D3D12_RESOURCE_BARRIER_TYPE_TRANSITION;
254 envBarrier.Transition.pResource = m_envCubemap->GetResource();
255 envBarrier.Transition.StateBefore = D3D12_RESOURCE_STATE_UNORDERED_ACCESS;
256 envBarrier.Transition.StateAfter = D3D12_RESOURCE_STATE_PIXEL_SHADER_RESOURCE;
257 envBarrier.Transition.Subresource = D3D12_RESOURCE_BARRIER_ALL_SUBRESOURCES;
258 cmdList->ResourceBarrier(1, &envBarrier);
259
260 // 出力先のテクスチャを作成
261 m_irradianceMap = std::make_unique<Texture>();
262 m_irradianceMap->InitializeAsCubemap(device, 32);
263
264 m_prefilterMap = std::make_unique<Texture>();
265 m_prefilterMap->InitializeAsCubemap(device, 128, 5);
266
267 m_brdfLUT = std::make_unique<Texture>();
268 m_brdfLUT->InitializeAsTexture2D(device, 512, 512, DXGI_FORMAT_R16G16_FLOAT);
269
270 // 事前計算の実行
271 builder.GenerateIrradianceMap(device, cmdList.Get(), m_envCubemap->GetCPUDescriptorHandle(), m_irradianceMap.get(), 32);
272 builder.GeneratePrefilterMap(device, cmdList.Get(), m_envCubemap->GetCPUDescriptorHandle(), m_prefilterMap.get(), 128);
273 builder.GenerateBRDFLUT(device, cmdList.Get(), m_brdfLUT.get(), 512);
274
275 // 生成した3つのIBLテクスチャのステートを UAV から SRV に遷移
276 D3D12_RESOURCE_BARRIER iblBarriers[3] = {};
277 // Irradiance
278 iblBarriers[0].Type = D3D12_RESOURCE_BARRIER_TYPE_TRANSITION;
279 iblBarriers[0].Transition.pResource = m_irradianceMap->GetResource();
280 iblBarriers[0].Transition.StateBefore = D3D12_RESOURCE_STATE_UNORDERED_ACCESS;
281 iblBarriers[0].Transition.StateAfter = D3D12_RESOURCE_STATE_PIXEL_SHADER_RESOURCE;
282 iblBarriers[0].Transition.Subresource = D3D12_RESOURCE_BARRIER_ALL_SUBRESOURCES;
283 // Prefilter
284 iblBarriers[1].Type = D3D12_RESOURCE_BARRIER_TYPE_TRANSITION;
285 iblBarriers[1].Transition.pResource = m_prefilterMap->GetResource();
286 iblBarriers[1].Transition.StateBefore = D3D12_RESOURCE_STATE_UNORDERED_ACCESS;
287 iblBarriers[1].Transition.StateAfter = D3D12_RESOURCE_STATE_PIXEL_SHADER_RESOURCE;
288 iblBarriers[1].Transition.Subresource = D3D12_RESOURCE_BARRIER_ALL_SUBRESOURCES;
289 // BRDF LUT
290 iblBarriers[2].Type = D3D12_RESOURCE_BARRIER_TYPE_TRANSITION;
291 iblBarriers[2].Transition.pResource = m_brdfLUT->GetResource();
292 iblBarriers[2].Transition.StateBefore = D3D12_RESOURCE_STATE_UNORDERED_ACCESS;
293 iblBarriers[2].Transition.StateAfter = D3D12_RESOURCE_STATE_PIXEL_SHADER_RESOURCE;
294 iblBarriers[2].Transition.Subresource = D3D12_RESOURCE_BARRIER_ALL_SUBRESOURCES;
295
296 cmdList->ResourceBarrier(3, iblBarriers);
297
298 // 6. コマンドリストを閉じて実行
299 cmdList->Close();
300 ID3D12CommandList* ppCommandLists[] = { cmdList.Get() };
301 queue->ExecuteCommandLists(1, ppCommandLists);
302
303 // GPUの完了を待機
304 ComPtr<ID3D12Fence> fence;
305 device->CreateFence(0, D3D12_FENCE_FLAG_NONE, IID_PPV_ARGS(&fence));
306 queue->Signal(fence.Get(), 1);
307
308 HANDLE fenceEvent = CreateEvent(nullptr, FALSE, FALSE, nullptr);
309 if (fence->GetCompletedValue() < 1)
310 {
311 fence->SetEventOnCompletion(1, fenceEvent);
312 WaitForSingleObject(fenceEvent, INFINITE);
313 }
314 CloseHandle(fenceEvent);
315
316 builder.Shutdown();
317
318 m_currentLoadedHDRI = filepath;
319 SPAN_LOG("Environment Cubemap generated successfully from %s", filepath.c_str());
320 return true;
321 }

323 void Renderer::BindTexture(ID3D12GraphicsCommandList* cmd, Texture* texture, uint32 rootIndex, D3D12_SRV_DIMENSION dimension)
324 {
325 if (!m_frameSrvHeap) return;
326
327 auto device = context->GetDevice();
328 D3D12_CPU_DESCRIPTOR_HANDLE destCpu = m_frameSrvHeap->GetCPUDescriptorHandleForHeapStart();
329 destCpu.ptr += m_frameSrvHeapOffset * m_srvDescriptorSize;
330
331 D3D12_GPU_DESCRIPTOR_HANDLE destGpu = m_frameSrvHeap->GetGPUDescriptorHandleForHeapStart();
332 destGpu.ptr += m_frameSrvHeapOffset * m_srvDescriptorSize;
333
334 if (texture && texture->GetSRVHeap())
335 {
336 // 通常のテクスチャ
337 device->CopyDescriptorsSimple(1, destCpu, texture->GetCPUDescriptorHandle(), D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV);
338 }
339 else if (texture && texture->GetResource())
340 {
341 // リソースはあるがSRVが無い場合、ここでSRVを作成する
342 D3D12_SHADER_RESOURCE_VIEW_DESC srvDesc = {};
343 srvDesc.Shader4ComponentMapping = D3D12_DEFAULT_SHADER_4_COMPONENT_MAPPING;
344 srvDesc.Format = texture->GetResource()->GetDesc().Format;
345 srvDesc.ViewDimension = dimension;
346 if (dimension == D3D12_SRV_DIMENSION_TEXTURECUBE) srvDesc.TextureCube.MipLevels = 1;
347 else if (dimension == D3D12_SRV_DIMENSION_TEXTURE2DARRAY) { srvDesc.Texture2DArray.MipLevels = 1; srvDesc.Texture2DArray.ArraySize = 1; }
348 else srvDesc.Texture2D.MipLevels = 1;
349
350 device->CreateShaderResourceView(texture->GetResource(), &srvDesc, destCpu);
351 }
352 else
353 {
354 // テクスチャが無い場合のダミー
355 D3D12_SHADER_RESOURCE_VIEW_DESC nullDesc = {};
356 nullDesc.Shader4ComponentMapping = D3D12_DEFAULT_SHADER_4_COMPONENT_MAPPING;
357 nullDesc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
358 nullDesc.ViewDimension = dimension;
359 if (dimension == D3D12_SRV_DIMENSION_TEXTURECUBE) nullDesc.TextureCube.MipLevels = 1;
360 else if (dimension == D3D12_SRV_DIMENSION_TEXTURE2DARRAY) { nullDesc.Texture2DArray.MipLevels = 1; nullDesc.Texture2DArray.ArraySize = 1; }
361 else nullDesc.Texture2D.MipLevels = 1;
362
363 device->CreateShaderResourceView(nullptr, &nullDesc, destCpu);
364 }
365
366 cmd->SetGraphicsRootDescriptorTable(rootIndex, destGpu);
367 m_frameSrvHeapOffset++;
368 }

463 bool Renderer::CreateRootSignature()
464 {
465 D3D12_ROOT_PARAMETER rootParameters[16];
466
467 // [0] Transform (b0)
468 rootParameters[0].ParameterType = D3D12_ROOT_PARAMETER_TYPE_CBV;
469 rootParameters[0].Descriptor.ShaderRegister = 0;
470 rootParameters[0].Descriptor.RegisterSpace = 0;
471 rootParameters[0].ShaderVisibility = D3D12_SHADER_VISIBILITY_ALL;
472
473 // [1] Material (b1)
474 rootParameters[1].ParameterType = D3D12_ROOT_PARAMETER_TYPE_CBV;
475 rootParameters[1].Descriptor.ShaderRegister = 1;
476 rootParameters[1].Descriptor.RegisterSpace = 0;
477 rootParameters[1].ShaderVisibility = D3D12_SHADER_VISIBILITY_ALL;
478
479 D3D12_DESCRIPTOR_RANGE ranges[13] = {};
480 for (int i = 0; i < 13; i++)
481 {
482 ranges[i].RangeType = D3D12_DESCRIPTOR_RANGE_TYPE_SRV;
483 ranges[i].NumDescriptors = 1;
484 ranges[i].BaseShaderRegister = i;
485 ranges[i].RegisterSpace = 0;
486 ranges[i].OffsetInDescriptorsFromTableStart = D3D12_DESCRIPTOR_RANGE_OFFSET_APPEND;
487
488 rootParameters[2 + i].ParameterType = D3D12_ROOT_PARAMETER_TYPE_DESCRIPTOR_TABLE;
489 rootParameters[2 + i].DescriptorTable.NumDescriptorRanges = 1;
490 rootParameters[2 + i].DescriptorTable.pDescriptorRanges = &ranges[i];
491 rootParameters[2 + i].ShaderVisibility = D3D12_SHADER_VISIBILITY_ALL;
492 }
493
494 // LightBuffer (b2)
495 rootParameters[15].ParameterType = D3D12_ROOT_PARAMETER_TYPE_CBV;
496 rootParameters[15].Descriptor.ShaderRegister = 2;
497 rootParameters[15].Descriptor.RegisterSpace = 0;
498 rootParameters[15].ShaderVisibility = D3D12_SHADER_VISIBILITY_PIXEL;
499
500 // --- Samplers ---
501 // s0: 通常のテクスチャサンプラー
