92 void Renderer::DrawMesh(Mesh* mesh, Material* material, const Matrix4x4& worldMatrix)
93 {
94 if (!mesh || !material || !commandList) return;
95
96 if (constantBufferIndex >= MAX_OBJECTS)
97 {
98 // 簡易的な上限チェック (本来は動的確保すべき)
99 return;
100 }
101
102 // 1. Transform更新
103 Matrix4x4 mvp = worldMatrix * viewMatrix * projectionMatrix;
104 TransformData data;
105 data.MVP.FromXM(XMMatrixTranspose(mvp.ToXM()));
106 data.World.FromXM(XMMatrixTranspose(worldMatrix.ToXM()));
107
108 UINT8* dest = mappedConstantBuffer + (constantBufferIndex * CB_OBJ_SIZE);
109 memcpy(dest, &data, sizeof(TransformData));
110
111 // 2. Material更新
112 material->Update();
113
114 // 3. コマンドセット
115 commandList->SetPipelineState(material->IsTransparent() ? pipelineStateTransparent.Get() : pipelineState.Get());
116 commandList->SetGraphicsRootSignature(rootSignature.Get());
117
118 // スロット0: Transform CBV
119 D3D12_GPU_VIRTUAL_ADDRESS cbAddress = constantBuffer->GetGPUVirtualAddress();
120 cbAddress += constantBufferIndex * CB_OBJ_SIZE;
121 commandList->SetGraphicsRootConstantBufferView(0, cbAddress);
122
123 // スロット1: Material CBV
124 commandList->SetGraphicsRootConstantBufferView(1, material->GetGPUVirtualAddress());
125
126 // スロット2: Texture SRV
127 if (material->GetTexture())
128 {
129 ID3D12DescriptorHeap* ppHeaps[] = { material->GetTexture()->GetSRVHeap() };
130 commandList->SetDescriptorHeaps(1, ppHeaps);
131 commandList->SetGraphicsRootDescriptorTable(2, material->GetTexture()->GetSRVHeap()->GetGPUDescriptorHandleForHeapStart());
132 }
133
134 mesh->Draw(commandList);
135
136 constantBufferIndex++;
137 }

145 bool Renderer::CreateRootSignature()
146 {
147 // ルートパラメータ定義 (CBV x2, Table x1)
148 D3D12_ROOT_PARAMETER rootParameters[3];
149
150 // [0] Transform (b0)
151 rootParameters[0].ParameterType = D3D12_ROOT_PARAMETER_TYPE_CBV;
152 rootParameters[0].Descriptor.ShaderRegister = 0;
153 rootParameters[0].Descriptor.RegisterSpace = 0;
154 rootParameters[0].ShaderVisibility = D3D12_SHADER_VISIBILITY_VERTEX;
155
156 // [1] Material (b1)
157 rootParameters[1].ParameterType = D3D12_ROOT_PARAMETER_TYPE_CBV;
158 rootParameters[1].Descriptor.ShaderRegister = 1;
159 rootParameters[1].Descriptor.RegisterSpace = 0;
160 rootParameters[1].ShaderVisibility = D3D12_SHADER_VISIBILITY_PIXEL;
161
162 // [2] Texture Table (t0)
163 D3D12_DESCRIPTOR_RANGE descriptorRange = {};
164 descriptorRange.RangeType = D3D12_DESCRIPTOR_RANGE_TYPE_SRV;
165 descriptorRange.NumDescriptors = 1;
166 descriptorRange.BaseShaderRegister = 0;
167 descriptorRange.RegisterSpace = 0;
168 descriptorRange.OffsetInDescriptorsFromTableStart = D3D12_DESCRIPTOR_RANGE_OFFSET_APPEND;
169
170 rootParameters[2].ParameterType = D3D12_ROOT_PARAMETER_TYPE_DESCRIPTOR_TABLE;
171 rootParameters[2].DescriptorTable.NumDescriptorRanges = 1;
172 rootParameters[2].DescriptorTable.pDescriptorRanges = &descriptorRange;
173 rootParameters[2].ShaderVisibility = D3D12_SHADER_VISIBILITY_PIXEL;
174
175 // サンプラー (Static Sampler)
176 D3D12_STATIC_SAMPLER_DESC sampler = {};
177 sampler.Filter = D3D12_FILTER_MIN_MAG_MIP_LINEAR;
178 sampler.AddressU = D3D12_TEXTURE_ADDRESS_MODE_WRAP;
179 sampler.AddressV = D3D12_TEXTURE_ADDRESS_MODE_WRAP;
180 sampler.AddressW = D3D12_TEXTURE_ADDRESS_MODE_WRAP;
181 sampler.MipLODBias = 0;
182 sampler.MaxAnisotropy = 0;
183 sampler.ComparisonFunc = D3D12_COMPARISON_FUNC_NEVER;
184 sampler.BorderColor = D3D12_STATIC_BORDER_COLOR_TRANSPARENT_BLACK;
185 sampler.MinLOD = 0.0f;
186 sampler.MaxLOD = D3D12_FLOAT32_MAX;
187 sampler.ShaderRegister = 0;
188 sampler.RegisterSpace = 0;
189 sampler.ShaderVisibility = D3D12_SHADER_VISIBILITY_PIXEL;
190
191 D3D12_ROOT_SIGNATURE_DESC rootSignatureDesc = {};
192 rootSignatureDesc.NumParameters = _countof(rootParameters);
193 rootSignatureDesc.pParameters = rootParameters;
194 rootSignatureDesc.NumStaticSamplers = 1;
195 rootSignatureDesc.pStaticSamplers = &sampler;
196 rootSignatureDesc.Flags = D3D12_ROOT_SIGNATURE_FLAG_ALLOW_INPUT_ASSEMBLER_INPUT_LAYOUT;
197
198 ComPtr<ID3DBlob> signature;
199 ComPtr<ID3DBlob> error;
200 if (FAILED(D3D12SerializeRootSignature(&rootSignatureDesc, D3D_ROOT_SIGNATURE_VERSION_1, &signature, &error)))
201 {
202 SPAN_ERROR("RootSignature Serialize Failed: %s", (char*)error->GetBufferPointer());
203 return false;
204 }
205
206 if (FAILED(context->GetDevice()->CreateRootSignature(0, signature->GetBufferPointer(), signature->GetBufferSize(), IID_PPV_ARGS(&rootSignature))))
207 {
208 return false;
209 }
210
211 return true;
212 }
