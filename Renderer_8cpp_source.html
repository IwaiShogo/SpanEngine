26 bool Renderer::Initialize(GraphicsContext* inContext)
27 {
28 this->context = inContext;
29 if (!context) return false;
30 auto device = context->GetDevice();
31
32 if (FAILED(device->CreateFence(0, D3D12_FENCE_FLAG_NONE, IID_PPV_ARGS(&m_waitFence)))) return false;
33 m_waitEvent = CreateEvent(nullptr, FALSE, FALSE, nullptr);
34 m_waitFenceValue = 1;
35
36 if (!CreateRootSignature()) return false;
37
38 vs = new Shader(); if (!vs->Load(L"Basic.hlsl", ShaderType::Vertex, "VSMain")) return false;
39 ps = new Shader(); if (!ps->Load(L"Basic.hlsl", ShaderType::Pixel, "PSMain")) return false;
40
41 if (!CreatePipelineState()) return false;
42 if (!CreateConstantBuffer()) return false;
43
44 // ダミーDescriptorの作成
45 if (!CreateDummyDescriptors()) return false;
46
47 // フレーム用の巨大な Descriptor Heap を作成
48 D3D12_DESCRIPTOR_HEAP_DESC heapDesc = {};
49 heapDesc.NumDescriptors = 4096;
50 heapDesc.Type = D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV;
51 heapDesc.Flags = D3D12_DESCRIPTOR_HEAP_FLAG_SHADER_VISIBLE;
52 if (FAILED(device->CreateDescriptorHeap(&heapDesc, IID_PPV_ARGS(&m_frameSrvHeap)))) return false;
53 m_srvDescriptorSize = device->GetDescriptorHandleIncrementSize(D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV);
54
55 // --- Passes Initialization ---
56 m_passManager = std::make_unique<RenderPassManager>();
57 if (!m_passManager->Initialize(context)) return false;
58
59 m_lightManager = std::make_unique<LightManager>();
60 if (!m_lightManager->Initialize(device)) return false;
61
62 SPAN_LOG("Renderer Initialized Successfully with Render Passes!");
63 return true;
64 }

85 D3D12_GPU_VIRTUAL_ADDRESS Renderer::AllocateCBV(const void* data, size_t sizeInBytes)
86 {
87 if (constantBufferIndex >= MAX_OBJECTS) return 0;
88
89 UINT8* dest = mappedConstantBuffer + (static_cast<SIZE_T>(constantBufferIndex) * CB_OBJ_SIZE);
90 memcpy(dest, data, sizeInBytes);
91
92 D3D12_GPU_VIRTUAL_ADDRESS addr = constantBuffer->GetGPUVirtualAddress() + (static_cast<SIZE_T>(constantBufferIndex) * CB_OBJ_SIZE);
93 constantBufferIndex++;
94
95 return addr;

138 void Renderer::BindGlobalResources()
139 {
140 if (!commandList) return;
141
142 // 他のパスで書き換えられたルートシグネチャとヒープをメインパス用に直す
143 ID3D12DescriptorHeap* heaps[] = { m_frameSrvHeap.Get() };

146
147 // Light Manager (b2 -> Register 19)

153 BindShadowMap(commandList, m_passManager->GetPointShadowPass() ? m_passManager->GetPointShadowPass()->GetShadowMap() : nullptr, 10, D3D12_SRV_DIMENSION_TEXTURECUBE);
154
155 // IBL Textures (t11 ~ t13) + OpaqueTexture(t14)
156 BindTexture(commandList, m_irradianceMap.get(), 11, D3D12_SRV_DIMENSION_TEXTURECUBE);
157 BindTexture(commandList, m_prefilterMap.get(), 12, D3D12_SRV_DIMENSION_TEXTURECUBE);
158 BindTexture(commandList, m_brdfLUT.get(), 13, D3D12_SRV_DIMENSION_TEXTURE2D);
159 BindTexture(commandList, m_opaqueCaptureTex.get(), 14, D3D12_SRV_DIMENSION_TEXTURE2D);
160
161 if (m_passManager->GetSSAOBlurPass() && m_passManager->GetSSAOBlurPass()->GetBlurredSSAOMap())
162 {
163 BindRenderTargetSRV(commandList, m_passManager->GetSSAOBlurPass()->GetBlurredSSAOMap(), 15);
164 }
165 else
166 {
167 BindTexture(commandList, nullptr, 15, D3D12_SRV_DIMENSION_TEXTURE2D);
168 }
169
170 BindComputeBufferSRV(commandList, m_lightManager ? m_lightManager->GetLightDataBuffer() : nullptr, 16);
171 BindComputeBufferSRV(commandList, m_lightManager ? m_lightManager->GetLightGrid() : nullptr, 17);
172 BindComputeBufferSRV(commandList, m_lightManager ? m_lightManager->GetLightIndexList() : nullptr, 18);

175 void Renderer::DrawMesh(Mesh* mesh, Material* material, const Matrix4x4& worldMatrix)
176 {
177 if (!mesh || !material || !commandList) return;
178
179 Matrix4x4 mvp = worldMatrix * viewMatrix * projectionMatrix;
180 TransformData data;
181 data.MVP.FromXM(XMMatrixTranspose(mvp.ToXM()));
182 data.World.FromXM(XMMatrixTranspose(worldMatrix.ToXM()));
183
184 D3D12_GPU_VIRTUAL_ADDRESS cbAddr = AllocateCBV(&data, sizeof(TransformData));
185 if (cbAddr == 0) return;
186
187 material->Update();
188 commandList->SetPipelineState(material->GetBlendMode() == BlendMode::Transparent ? pipelineStateTransparent.Get() : pipelineState.Get());
189 commandList->SetGraphicsRootSignature(rootSignature.Get());
190
191 commandList->SetGraphicsRootConstantBufferView(0, cbAddr);
192 commandList->SetGraphicsRootConstantBufferView(1, material->GetGPUVirtualAddress());
193
194 // PBR Textures (t0 ~ t5)
195 Texture* textures[6] = { material->GetAlbedoMap(), material->GetNormalMap(), material->GetMetallicMap(), material->GetRoughnessMap(), material->GetAOMap(), material->GetEmissiveMap() };
196 for (int i = 0; i < 6; i++) {
197 BindTexture(commandList, textures[i], 2 + i, D3D12_SRV_DIMENSION_TEXTURE2D);
198 }
199
200 mesh->Draw(commandList);
201 }

211 bool Renderer::LoadEnvironmentMap(const std::string& filepath)
212 {
213 if (filepath.empty() || m_currentLoadedHDRI == filepath) return true;
214
215 auto device = context->GetDevice();
216 auto queue = context->GetCommandQueue();
217
218 // 1. パノラマHDRI画像を読み込む (.hdr)
219 Texture panoramaTex;
220 if (!panoramaTex.Initialize(device, queue, filepath))
221 {
222 SPAN_ERROR("Failed to load HDRI: %s", filepath.c_str());
223 return false;
224 }
225
226 // 2. 出力先の空の Cubemap を作成
227 m_envCubemap = std::make_unique<Texture>();
228 if (!m_envCubemap->InitializeAsCubemap(device, 1024)) return false;
229
230 // 3. Builderの初期化
231 IBLBuilder builder;
232 if (!builder.Initialize(device)) return false;
233
234 // 4. Compute Shader用の1回限りのコマンドリストを作成
235 ComPtr<ID3D12CommandAllocator> cmdAlloc;
236 ComPtr<ID3D12GraphicsCommandList> cmdList;
237 device->CreateCommandAllocator(D3D12_COMMAND_LIST_TYPE_DIRECT, IID_PPV_ARGS(&cmdAlloc));
238 device->CreateCommandList(0, D3D12_COMMAND_LIST_TYPE_DIRECT, cmdAlloc.Get(), nullptr, IID_PPV_ARGS(&cmdList));
239
240 // 5. 変換処理
241 builder.GenerateCubemapFromPanorama(device, cmdList.Get(), panoramaTex.GetCPUDescriptorHandle(), m_envCubemap.get(), 1024);
242
243 // m_envCubemap のステートを UAV から SRV に遷移
244 D3D12_RESOURCE_BARRIER envBarrier = {};
245 envBarrier.Type = D3D12_RESOURCE_BARRIER_TYPE_TRANSITION;
246 envBarrier.Transition.pResource = m_envCubemap->GetResource();
247 envBarrier.Transition.StateBefore = D3D12_RESOURCE_STATE_UNORDERED_ACCESS;
248 envBarrier.Transition.StateAfter = D3D12_RESOURCE_STATE_PIXEL_SHADER_RESOURCE;
249 envBarrier.Transition.Subresource = D3D12_RESOURCE_BARRIER_ALL_SUBRESOURCES;
250 cmdList->ResourceBarrier(1, &envBarrier);
251
252 // 出力先のテクスチャを作成
253 m_irradianceMap = std::make_unique<Texture>();
254 m_irradianceMap->InitializeAsCubemap(device, 32);
255
256 m_prefilterMap = std::make_unique<Texture>();
257 m_prefilterMap->InitializeAsCubemap(device, 128, 5);
258
259 m_brdfLUT = std::make_unique<Texture>();
260 m_brdfLUT->InitializeAsTexture2D(device, 512, 512, DXGI_FORMAT_R16G16_FLOAT);
261
262 // 事前計算の実行
263 builder.GenerateIrradianceMap(device, cmdList.Get(), m_envCubemap->GetCPUDescriptorHandle(), m_irradianceMap.get(), 32);
264 builder.GeneratePrefilterMap(device, cmdList.Get(), m_envCubemap->GetCPUDescriptorHandle(), m_prefilterMap.get(), 128);
265 builder.GenerateBRDFLUT(device, cmdList.Get(), m_brdfLUT.get(), 512);
266
267 // 生成した3つのIBLテクスチャのステートを UAV から SRV に遷移
268 D3D12_RESOURCE_BARRIER iblBarriers[3] = {};
269 // Irradiance
270 iblBarriers[0].Type = D3D12_RESOURCE_BARRIER_TYPE_TRANSITION;
271 iblBarriers[0].Transition.pResource = m_irradianceMap->GetResource();
272 iblBarriers[0].Transition.StateBefore = D3D12_RESOURCE_STATE_UNORDERED_ACCESS;
273 iblBarriers[0].Transition.StateAfter = D3D12_RESOURCE_STATE_PIXEL_SHADER_RESOURCE;
274 iblBarriers[0].Transition.Subresource = D3D12_RESOURCE_BARRIER_ALL_SUBRESOURCES;
275 // Prefilter
276 iblBarriers[1].Type = D3D12_RESOURCE_BARRIER_TYPE_TRANSITION;
277 iblBarriers[1].Transition.pResource = m_prefilterMap->GetResource();
278 iblBarriers[1].Transition.StateBefore = D3D12_RESOURCE_STATE_UNORDERED_ACCESS;
279 iblBarriers[1].Transition.StateAfter = D3D12_RESOURCE_STATE_PIXEL_SHADER_RESOURCE;
280 iblBarriers[1].Transition.Subresource = D3D12_RESOURCE_BARRIER_ALL_SUBRESOURCES;
281 // BRDF LUT
282 iblBarriers[2].Type = D3D12_RESOURCE_BARRIER_TYPE_TRANSITION;
283 iblBarriers[2].Transition.pResource = m_brdfLUT->GetResource();
284 iblBarriers[2].Transition.StateBefore = D3D12_RESOURCE_STATE_UNORDERED_ACCESS;
285 iblBarriers[2].Transition.StateAfter = D3D12_RESOURCE_STATE_PIXEL_SHADER_RESOURCE;
286 iblBarriers[2].Transition.Subresource = D3D12_RESOURCE_BARRIER_ALL_SUBRESOURCES;
287
288 cmdList->ResourceBarrier(3, iblBarriers);
289
290 // 6. コマンドリストを閉じて実行
291 cmdList->Close();
292 ID3D12CommandList* ppCommandLists[] = { cmdList.Get() };
293 queue->ExecuteCommandLists(1, ppCommandLists);
294
295 // GPUの完了を待機
296 ComPtr<ID3D12Fence> fence;
297 device->CreateFence(0, D3D12_FENCE_FLAG_NONE, IID_PPV_ARGS(&fence));
298 queue->Signal(fence.Get(), 1);
299
300 HANDLE fenceEvent = CreateEvent(nullptr, FALSE, FALSE, nullptr);
301 if (fenceEvent)
302 {
303 if (fence->GetCompletedValue() < 1)
304 {
305 fence->SetEventOnCompletion(1, fenceEvent);
306 WaitForSingleObject(fenceEvent, INFINITE);
307 }
308 CloseHandle(fenceEvent);
309 }
310
311 builder.Shutdown();
312
313 m_currentLoadedHDRI = filepath;
314 SPAN_LOG("Environment Cubemap generated successfully from %s", filepath.c_str());
315 return true;
316 }

318 void Renderer::BindTexture(ID3D12GraphicsCommandList* cmd, Texture* texture, uint32 rootIndex, D3D12_SRV_DIMENSION dimension)
319 {
320 if (!m_frameSrvHeap) return;
321
322 auto device = context->GetDevice();
323 D3D12_CPU_DESCRIPTOR_HANDLE destCpu = m_frameSrvHeap->GetCPUDescriptorHandleForHeapStart();
324 destCpu.ptr += static_cast<SIZE_T>(m_frameSrvHeapOffset) * m_srvDescriptorSize;
325
326 D3D12_GPU_DESCRIPTOR_HANDLE destGpu = m_frameSrvHeap->GetGPUDescriptorHandleForHeapStart();
327 destGpu.ptr += static_cast<SIZE_T>(m_frameSrvHeapOffset) * m_srvDescriptorSize;
328
329 if (texture && texture->GetSRVHeap())
330 {
331 // 通常のテクスチャ
332 device->CopyDescriptorsSimple(1, destCpu, texture->GetCPUDescriptorHandle(), D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV);
333 }
334 else if (texture && texture->GetResource())
335 {
336 // リソースはあるがSRVヒープが無い場合、ここでSRVを作成する
337 D3D12_SHADER_RESOURCE_VIEW_DESC srvDesc = {};
338 srvDesc.Shader4ComponentMapping = D3D12_DEFAULT_SHADER_4_COMPONENT_MAPPING;
339 srvDesc.Format = texture->GetResource()->GetDesc().Format;
340 srvDesc.ViewDimension = dimension;
341 if (dimension == D3D12_SRV_DIMENSION_TEXTURECUBE) srvDesc.TextureCube.MipLevels = 1;
342 else if (dimension == D3D12_SRV_DIMENSION_TEXTURE2DARRAY) { srvDesc.Texture2DArray.MipLevels = 1; srvDesc.Texture2DArray.ArraySize = 1; }
343 else srvDesc.Texture2D.MipLevels = 1;
344
345 device->CreateShaderResourceView(texture->GetResource(), &srvDesc, destCpu);
346 }
347 else
348 {
349 // テクスチャが無い場合のダミー（事前生成した安全なキャッシュを使用）
350 device->CopyDescriptorsSimple(1, destCpu, GetDummyDescriptor(dimension), D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV);
351 }
352
353 cmd->SetGraphicsRootDescriptorTable(rootIndex, destGpu);
354 m_frameSrvHeapOffset++;
355 }

484 void Renderer::CaptureOpaqueBackground(ID3D12Resource* currentRenderTarget)
485 {
486 if (!commandList || !currentRenderTarget || !m_opaqueCaptureTex || !m_opaqueCaptureTex->GetResource()) return;
487
488 D3D12_RESOURCE_BARRIER barriers[2] = {};
489
490 // レンダーターゲットを COPY_SOURCE に、キャプチャ用テクスチャを COPY_DEST に状態遷移
491 barriers[0].Type = D3D12_RESOURCE_BARRIER_TYPE_TRANSITION;
492 barriers[0].Transition.pResource = currentRenderTarget;
493 barriers[0].Transition.StateBefore = D3D12_RESOURCE_STATE_RENDER_TARGET;
494 barriers[0].Transition.StateAfter = D3D12_RESOURCE_STATE_COPY_SOURCE;
