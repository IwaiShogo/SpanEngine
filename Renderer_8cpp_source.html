142 void Renderer::DrawMesh(Mesh* mesh, Material* material, const Matrix4x4& worldMatrix)
143 {
144 if (!mesh || !material || !commandList) return;
145
146 if (constantBufferIndex >= MAX_OBJECTS)
147 {
148 // 簡易的な上限チェック (本来は動的確保すべき)
149 return;
150 }
151
152 // 1. Transform更新
153 Matrix4x4 mvp = worldMatrix * viewMatrix * projectionMatrix;
154 TransformData data;
155 data.MVP.FromXM(XMMatrixTranspose(mvp.ToXM()));
156 data.World.FromXM(XMMatrixTranspose(worldMatrix.ToXM()));
157
158 UINT8* dest = mappedConstantBuffer + (constantBufferIndex * CB_OBJ_SIZE);
159 memcpy(dest, &data, sizeof(TransformData));
160
161 // 2. Material更新
162 material->Update();
163
164 // 3. コマンドセット
165 commandList->SetPipelineState(material->IsTransparent() ? pipelineStateTransparent.Get() : pipelineState.Get());
166 commandList->SetGraphicsRootSignature(rootSignature.Get());
167
168 // スロット0: Transform CBV
169 D3D12_GPU_VIRTUAL_ADDRESS cbAddress = constantBuffer->GetGPUVirtualAddress();
170 cbAddress += constantBufferIndex * CB_OBJ_SIZE;
171 commandList->SetGraphicsRootConstantBufferView(0, cbAddress);
172
173 // スロット1: Material CBV
174 commandList->SetGraphicsRootConstantBufferView(1, material->GetGPUVirtualAddress());
175
176 // スロット2: Texture SRV
177 if (material->GetTexture())
178 {
179 ID3D12DescriptorHeap* ppHeaps[] = { material->GetTexture()->GetSRVHeap() };
180 commandList->SetDescriptorHeaps(1, ppHeaps);
181 commandList->SetGraphicsRootDescriptorTable(2, material->GetTexture()->GetSRVHeap()->GetGPUDescriptorHandleForHeapStart());
182 }
183
184 mesh->Draw(commandList);
185
186 constantBufferIndex++;

195 void Renderer::WaitForGPU()
196 {
197 if (!context || !context->GetCommandQueue() || !m_waitFence) return;
198
199 ID3D12CommandQueue* queue = context->GetCommandQueue();
200
201 // 次の値をシグナル
202 const uint64_t fenceToWaitFor = m_waitFenceValue;
203 queue->Signal(m_waitFence.Get(), fenceToWaitFor);
204 m_waitFenceValue++;
205
206 // GPUがここまで到達するのを待つ
207 if (m_waitFence->GetCompletedValue() < fenceToWaitFor)
208 {
209 m_waitFence->SetEventOnCompletion(fenceToWaitFor, m_waitEvent);
210 WaitForSingleObject(m_waitEvent, INFINITE);
211 }
212 }
