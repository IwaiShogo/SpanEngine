78 void ShadowPass::BeginPass(ID3D12GraphicsCommandList* cmd)
79 {
80 if (!m_shadowMap || !cmd) return;
81 D3D12_RESOURCE_BARRIER barrier = {};
82 barrier.Type = D3D12_RESOURCE_BARRIER_TYPE_TRANSITION;
83 barrier.Transition.pResource = m_shadowMap->GetResource();
84 barrier.Transition.StateBefore = D3D12_RESOURCE_STATE_PIXEL_SHADER_RESOURCE;
85 barrier.Transition.StateAfter = D3D12_RESOURCE_STATE_DEPTH_WRITE;
86 barrier.Transition.Subresource = D3D12_RESOURCE_BARRIER_ALL_SUBRESOURCES;
87 cmd->ResourceBarrier(1, &barrier);
88
89 cmd->SetPipelineState(m_pso.Get());
90 cmd->SetGraphicsRootSignature(m_rootSignature.Get());
91 }

93 void ShadowPass::SetRenderTarget(ID3D12GraphicsCommandList* cmd, uint32 sliceIndex)
94 {
95 if (!m_shadowMap || !cmd) return;
96 D3D12_CPU_DESCRIPTOR_HANDLE dsvHandle = m_shadowMap->GetDSV(sliceIndex);
97 cmd->OMSetRenderTargets(0, nullptr, FALSE, &dsvHandle);
98 cmd->ClearDepthStencilView(dsvHandle, D3D12_CLEAR_FLAG_DEPTH, 1.0f, 0, 0, nullptr);
99
100 D3D12_VIEWPORT vp = { 0.0f, 0.0f, (float)m_shadowMap->GetWidth(), (float)m_shadowMap->GetHeight(), 0.0f, 1.0f };
101 D3D12_RECT scissor = { 0, 0, (long)m_shadowMap->GetWidth(), (long)m_shadowMap->GetHeight() };
102 cmd->RSSetViewports(1, &vp);
103 cmd->RSSetScissorRects(1, &scissor);

118 void ShadowPass::DrawMesh(Renderer* renderer, ID3D12GraphicsCommandList* cmd, Mesh* mesh, const Matrix4x4& worldMatrix, const Matrix4x4& lightSpaceMatrix)
119 {
120 if (!mesh || !cmd) return;
121
122 TransformData data;
123 Matrix4x4 mvp = worldMatrix * lightSpaceMatrix;
124 data.MVP.FromXM(XMMatrixTranspose(mvp.ToXM()));
125 data.World.FromXM(XMMatrixTranspose(worldMatrix.ToXM()));
126
127 D3D12_GPU_VIRTUAL_ADDRESS cbAddr = renderer->AllocateCBV(&data, sizeof(TransformData));
128 if (cbAddr == 0) return;
129
130 cmd->SetGraphicsRootConstantBufferView(0, cbAddr);
131 mesh->Draw(cmd);
132 }
