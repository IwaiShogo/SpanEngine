83 void SceneViewPanel::DrawGizmo(const Vector2& pos, const Vector2& size)
84 {
85 World& world = Application::Get().GetWorld();
86
87 // カメラ情報の取得
88 Entity cameraEntity = Entity::Null;
89 Matrix4x4 cameraView = Matrix4x4::Identity();
90 Matrix4x4 cameraProj = Matrix4x4::Identity();
91
92 // シーン内のアクティブなカメラ(EditorCamera優先)を探す
93 world.ForEach<Camera, LocalToWorld>([&](Entity e, Camera& cam, LocalToWorld& ltw)
94 {
95 // EditorCameraコンポーネントを持っていればそれを優先採用
96 if (world.HasComponent<EditorCamera>(e) || cameraEntity.IsNull())
97 {
98 cameraEntity = e;
99 cameraView = ltw.Value.Invert();
100
101 // アスペクト比を現在のビューポートに合わせる
102 float aspect = size.x / size.y;
103 if (aspect <= 0.0f) aspect = 1.0f;
104
105 cameraProj = Matrix4x4::PerspectiveFovLH(ToRadians(cam.Fov), aspect, cam.NearClip, cam.FarClip);
106 }
107 });
108
109 if (cameraEntity.IsNull()) return;
110
111 ImGuizmo::SetOrthographic(false);
112
113 // --- View Cube (シーンギズモ) ---
114 // 右上に表示
115 {
116 Matrix4x4 viewManipulateMatrix = cameraView;
117 float viewSize = 128.0f;
118
119 ImGuizmo::ViewManipulate(
120 (float*)&viewManipulateMatrix,
121 5.0f,
122 ImVec2(pos.x + size.x - viewSize, pos.y),
123 ImVec2(viewSize, viewSize),
124 0x10101010
125 );
126
127 // ViewManipulateで操作された場合、カメラの行列を更新する必要がある
128 }
129
130 // --- 選択オブジェクトの操作ギズモ ---
131 Entity selectedEntity = SelectionManager::GetPrimary();
132 if (!selectedEntity.IsNull() && world.IsAlive(selectedEntity) && m_GizmoType != -1)
133 {
134 Transform* tc = world.GetComponentPtr<Transform>(selectedEntity);
135 if (tc)
136 {
137 // グリッド描画
138 if (m_GizmoType == ImGuizmo::OPERATION::TRANSLATE)
139 {
140 // ImGuizmo::DrawGrid((float*)&cameraView, (float*)&cameraProj, (float*)&Matrix4x4::Identity(), 100.0f);
141 }
142
143 // 現在のTransform行列
144 Matrix4x4 transformMatrix = Matrix4x4::TRS(tc->Position, tc->Rotation, tc->Scale);
145
146 // スナップ設定
147 float snapValue = 0.5f;
148 if (m_GizmoType == ImGuizmo::OPERATION::ROTATE) snapValue = m_SnapValueRotate;
149 else if (m_GizmoType == ImGuizmo::OPERATION::SCALE) snapValue = m_SnapValueScale;
150 else snapValue = m_SnapValueMove;
151
152 float snapValues[3] = { snapValue, snapValue, snapValue };
153
154 // ギズモ操作実行
155 ImGuizmo::Manipulate(
156 (float*)&cameraView,
157 (float*)&cameraProj,
158 (ImGuizmo::OPERATION)m_GizmoType,
159 (ImGuizmo::MODE)m_GizmoMode,
160 (float*)&transformMatrix,
161 nullptr,
162 m_UseSnap ? snapValues : nullptr
163 );
164
165 // 操作された場合、値を書き戻す
166 if (ImGuizmo::IsUsing())
167 {
168 Vector3 translation, rotation, scale;
169 ImGuizmo::DecomposeMatrixToComponents((float*)&transformMatrix, &translation.x, &rotation.x, &scale.x);
170
171 tc->Position = translation;
172 tc->Rotation = Quaternion::FromEuler(ToRadians(rotation.x), ToRadians(rotation.y), ToRadians(rotation.z));
173 tc->Scale = scale;
174 }
175 }
176 }
177
178 // --- ショートカットキー処理 ---
179 // シーンビューにフォーカスがある時のみ
180 if (ImGui::IsWindowFocused() && !ImGuizmo::IsUsing())
181 {
182 if (Input::GetKeyDown(Key::Q)) m_GizmoType = -1; // なし (選択モード)
183 if (Input::GetKeyDown(Key::W)) m_GizmoType = ImGuizmo::OPERATION::TRANSLATE;
184 if (Input::GetKeyDown(Key::E)) m_GizmoType = ImGuizmo::OPERATION::ROTATE;
185 if (Input::GetKeyDown(Key::R)) m_GizmoType = ImGuizmo::OPERATION::SCALE;
186 }
187 }

263 void SceneViewPanel::CalculateImageArea(const Vector2& avail, Vector2& outPos, Vector2& outSize)
264 {
265 // アスペクト比の倍率計算 (W / H)
266 float targetAspect = 0.0f;
267 switch (m_AspectRatio)
268 {
269 case AspectRatioType::Ratio_16_9: targetAspect = 16.0f / 9.0f; break;
270 case AspectRatioType::Ratio_16_10: targetAspect = 16.0f / 10.0f; break;
271 case AspectRatioType::Ratio_4_3: targetAspect = 4.0f / 3.0f; break;
272 case AspectRatioType::Ratio_21_9: targetAspect = 21.0f / 9.0f; break;
273 case AspectRatioType::Free: default:
274 outPos = { 0.0f, 0.0f };
275 outSize = avail;
276 return;
277 }
278
279 float currentAspect = avail.x / avail.y;
280
281 if (currentAspect > targetAspect)
282 {
283 // 横に長い -> 左右に黒帯 (Pillarbox)
284 outSize.y = avail.y;
285 outSize.x = outSize.y * targetAspect;
286 outPos.y = 0.0f;
287 outPos.x = (avail.x - outSize.x) * 0.5f;
288 }
289 else
290 {
291 // 縦に長い -> 上下に黒帯 (Letterbox)
292 outSize.x = avail.x;
293 outSize.y = outSize.x / targetAspect;
294 outPos.x = 0.0f;
295 outPos.y = (avail.y - outSize.y) * 0.5f;
296 }
297 }
