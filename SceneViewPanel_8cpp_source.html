97 void SceneViewPanel::DrawGizmo(const Vector2& pos, const Vector2& size)
98 {
99 // サイズが小さすぎる場合はギズモを描画しない
100 if (size.x < 1.0f || size.y < 1.0f) return;
101
102 World& world = Application::Get().GetWorld();
103
104 // カメラ情報の取得
105 Entity cameraEntity = Entity::Null;
106 Matrix4x4 cameraView = Matrix4x4::Identity();
107
108 float fov = 60.0f;
109 float nearClip = 0.1f;
110 float farClip = 1000.0f;
111
112 // シーン内のアクティブなカメラ(EditorCamera優先)を探す
113 world.ForEach<Camera, LocalToWorld>([&](Entity e, Camera& cam, LocalToWorld& ltw)
114 {
115 // EditorCameraコンポーネントを持っていればそれを優先採用
116 if (world.HasComponent<EditorCamera>(e) || cameraEntity.IsNull())
117 {
118 cameraEntity = e;
119 cameraView = ltw.Value.Invert();
120 cameraTransform = world.GetComponentPtr<Transform>(e);
121
122 fov = cam.Fov;
123 nearClip = cam.NearClip;
124 farClip = cam.FarClip;
125 }
126 });
127
128 if (cameraEntity.IsNull()) return;
129
130 ImGuizmo::SetOrthographic(false);
131
132 // 行列転置用バッファ
133 // 左手系 (LH) -> 右手系 (RH) への変換
134 Matrix4x4 viewRH = cameraView;
135 viewRH.m[0][2] *= -1.0f;
136 viewRH.m[1][2] *= -1.0f;
137 viewRH.m[2][2] *= -1.0f;
138 viewRH.m[3][2] *= -1.0f;
139
140 float aspect = size.x / size.y;
141 if (aspect <= 0.0f) aspect = 1.0f;
142
143 Matrix4x4 projRH;
144 projRH.FromXM(XMMatrixPerspectiveFovRH(ToRadians(fov), aspect, nearClip, farClip));
145
146 // 1. Scene Gizmo (ViewManipulate)
147 // ------------------------------------------------------------
148 {
149 float viewSize = 128.0f;
150 ImVec2 viewPos = ImVec2(pos.x + size.x - viewSize, pos.y);
151
152 ImGuizmo::ViewManipulate(
153 (float*)&viewRH,
154 8.0f,
155 viewPos,
156 ImVec2(viewSize, viewSize),
157 0x10101010
158 );
159
160 // 転置されたView行列を元に戻す
161 Matrix4x4 newViewLH = viewRH;
162 newViewLH.m[0][2] *= -1.0f;
163 newViewLH.m[1][2] *= -1.0f;
164 newViewLH.m[2][2] *= -1.0f;
165 newViewLH.m[3][2] *= -1.0f;
166
167 Matrix4x4 newCamWorld = newViewLH.Invert();
168
169 if (world.IsAlive(cameraEntity))
170 {
171 Transform* transform = world.GetComponentPtr<Transform>(cameraEntity);
172 if (transform)
173 {
174 Vector3 p, s;
175 Quaternion r;
176 if (newCamWorld.Decompose(p, r, s))
177 {
178 // 回転のみ適用
179 transform->Rotation = r;
180 }
181 }
182 }
183 }
184
185 // 2. Object Manipulate
186 // ------------------------------------------------------------
187 Entity selectedEntity = SelectionManager::GetPrimary();
188 if (!selectedEntity.IsNull() && world.IsAlive(selectedEntity) && m_GizmoType != -1)
189 {
190 Transform* tc = world.GetComponentPtr<Transform>(selectedEntity);
191 if (tc)
192 {
193 // 現在のTransform行列
194 Matrix4x4 transformMatrix = Matrix4x4::TRS(tc->Position, tc->Rotation, tc->Scale);
195
196 // オブジェクト行列も転置して渡す
197 Matrix4x4 objectMtx = transformMatrix;
198
199 // スナップ設定
200 float snapValue = 0.5f;
201 if (m_GizmoType == ImGuizmo::OPERATION::ROTATE) snapValue = m_SnapValueRotate;
202 else if (m_GizmoType == ImGuizmo::OPERATION::SCALE) snapValue = m_SnapValueScale;
203 else snapValue = m_SnapValueMove;
204
205 float snapValues[3] = { snapValue, snapValue, snapValue };
206
207 // IDを設定して競合を防ぐ
208 ImGuizmo::SetID((int)(uint64_t)selectedEntity.ID.Index);
209
210 // ギズモ操作実行
211 ImGuizmo::Manipulate(
212 (float*)&viewRH,
213 (float*)&projRH,
214 (ImGuizmo::OPERATION)m_GizmoType,
215 (ImGuizmo::MODE)m_GizmoMode,
216 (float*)&objectMtx,
217 nullptr,
218 m_UseSnap ? snapValues : nullptr
219 );
220
221 // 操作された場合、値を書き戻す
222 if (ImGuizmo::IsUsing())
223 {
224 // 結果を転置して戻す
225 Matrix4x4 newTransform = objectMtx;
226
227 Vector3 p, s;
228 Quaternion r;
229 if (newTransform.Decompose(p, r, s))
230 {
231 tc->Position = p;
232 tc->Rotation = r;
233 tc->Scale = s;
234 }
235 }
236 }
237 }
238
239 // ショートカットキー処理
240 if (ImGui::IsWindowFocused() && !ImGuizmo::IsUsing() && !Input::GetKey(Key::MouseRight))
241 {
242 if (Input::GetKeyDown(Key::Q)) m_GizmoType = -1; // なし (選択モード)
243 if (Input::GetKeyDown(Key::W)) m_GizmoType = ImGuizmo::OPERATION::TRANSLATE;
244 if (Input::GetKeyDown(Key::E)) m_GizmoType = ImGuizmo::OPERATION::ROTATE;
245 if (Input::GetKeyDown(Key::R)) m_GizmoType = ImGuizmo::OPERATION::SCALE;
246 }
247 }

323 void SceneViewPanel::CalculateImageArea(const Vector2& avail, Vector2& outPos, Vector2& outSize)
324 {
325 // アスペクト比の倍率計算 (W / H)
326 float targetAspect = 0.0f;
327 switch (m_AspectRatio)
328 {
329 case AspectRatioType::Ratio_16_9: targetAspect = 16.0f / 9.0f; break;
330 case AspectRatioType::Ratio_16_10: targetAspect = 16.0f / 10.0f; break;
331 case AspectRatioType::Ratio_4_3: targetAspect = 4.0f / 3.0f; break;
332 case AspectRatioType::Ratio_21_9: targetAspect = 21.0f / 9.0f; break;
333 case AspectRatioType::Free: default:
334 outPos = { 0.0f, 0.0f };
335 outSize = avail;
336 return;
337 }
338
339 float currentAspect = avail.x / avail.y;
340
341 if (currentAspect > targetAspect)
342 {
343 // 横に長い -> 左右に黒帯 (Pillarbox)
344 outSize.y = avail.y;
345 outSize.x = outSize.y * targetAspect;
346 outPos.y = 0.0f;
347 outPos.x = (avail.x - outSize.x) * 0.5f;
348 }
349 else
350 {
351 // 縦に長い -> 上下に黒帯 (Letterbox)
352 outSize.x = avail.x;
353 outSize.y = outSize.x / targetAspect;
354 outPos.x = 0.0f;
355 outPos.y = (avail.y - outSize.y) * 0.5f;
356 }
357 }
