138 bool Texture::UploadTexture(ID3D12Device* device, ID3D12CommandQueue* commandQueue,
139 const void* initialData, uint64_t w, uint64_t h, uint64_t bytesPerPixel, DXGI_FORMAT format)
140 {
141 // リソース記述
142 D3D12_RESOURCE_DESC textureDesc = {};
143 textureDesc.MipLevels = 1;
144 textureDesc.Format = format;
145 textureDesc.Width = w;
146 textureDesc.Height = h;
147 textureDesc.Flags = D3D12_RESOURCE_FLAG_NONE;
148 textureDesc.DepthOrArraySize = 1;
149 textureDesc.SampleDesc.Count = 1;
150 textureDesc.SampleDesc.Quality = 0;
151 textureDesc.Dimension = D3D12_RESOURCE_DIMENSION_TEXTURE2D;
152
153 // 1. デフォルトヒープ（VRAM）にテクスチャリソースを作成
154 D3D12_HEAP_PROPERTIES defaultHeap = { D3D12_HEAP_TYPE_DEFAULT };
155
156 if (FAILED(device->CreateCommittedResource(
157 &defaultHeap,
158 D3D12_HEAP_FLAG_NONE,
159 &textureDesc,
160 D3D12_RESOURCE_STATE_COPY_DEST,
161 nullptr,
162 IID_PPV_ARGS(&resource))))
163 {
164 return false;
165 }
166
167 // 2. アップロード用バッファのサイズ計算
168 uint64_t uploadBufferSize = 0;
169 D3D12_PLACED_SUBRESOURCE_FOOTPRINT footprint;
170 UINT64 rowSizeInBytes;
171 UINT numRows;
172 device->GetCopyableFootprints(&textureDesc, 0, 1, 0, &footprint, &numRows, &rowSizeInBytes, &uploadBufferSize);
173
174 // 3. アップロードヒープ（CPU書き込み可）のリソース作成
175 D3D12_HEAP_PROPERTIES uploadHeap = { D3D12_HEAP_TYPE_UPLOAD };
176 D3D12_RESOURCE_DESC bufferDesc = {};
177 bufferDesc.Dimension = D3D12_RESOURCE_DIMENSION_BUFFER;
178 bufferDesc.Alignment = 0;
179 bufferDesc.Width = uploadBufferSize;
180 bufferDesc.Height = 1;
181 bufferDesc.DepthOrArraySize = 1;
182 bufferDesc.MipLevels = 1;
183 bufferDesc.Format = DXGI_FORMAT_UNKNOWN;
184 bufferDesc.SampleDesc.Count = 1;
185 bufferDesc.Layout = D3D12_TEXTURE_LAYOUT_ROW_MAJOR;
186 bufferDesc.Flags = D3D12_RESOURCE_FLAG_NONE;
187
188 if (FAILED(device->CreateCommittedResource(
189 &uploadHeap,
190 D3D12_HEAP_FLAG_NONE,
191 &bufferDesc,
192 D3D12_RESOURCE_STATE_GENERIC_READ,
193 nullptr,
194 IID_PPV_ARGS(&uploadBuffer))))
195 {
196 return false;
197 }
198
199 // 4. データをアップロードバッファにコピー
200 void* mappedData = nullptr;
201 uploadBuffer->Map(0, nullptr, &mappedData);
202
203 const uint8_t* srcData = reinterpret_cast<const uint8_t*>(initialData);
204 uint8_t* destData = reinterpret_cast<uint8_t*>(mappedData);
205
206 for (UINT i = 0; i < h; ++i)
207 {
208 memcpy(destData + (i * footprint.Footprint.RowPitch),
209 srcData + (i * w * bytesPerPixel),
210 w * bytesPerPixel);
211 }
212 uploadBuffer->Unmap(0, nullptr);
213
214 // 5. コマンドリストを作ってコピー命令を発行
215 ComPtr<ID3D12CommandAllocator> cmdAlloc;
216 ComPtr<ID3D12GraphicsCommandList> cmdList;
217 device->CreateCommandAllocator(D3D12_COMMAND_LIST_TYPE_DIRECT, IID_PPV_ARGS(&cmdAlloc));
218 device->CreateCommandList(0, D3D12_COMMAND_LIST_TYPE_DIRECT, cmdAlloc.Get(), nullptr, IID_PPV_ARGS(&cmdList));
219
220 D3D12_TEXTURE_COPY_LOCATION srcLoc = {};
221 srcLoc.pResource = uploadBuffer.Get();
222 srcLoc.Type = D3D12_TEXTURE_COPY_TYPE_PLACED_FOOTPRINT;
223 srcLoc.PlacedFootprint = footprint;
224
225 D3D12_TEXTURE_COPY_LOCATION dstLoc = {};
226 dstLoc.pResource = resource.Get();
227 dstLoc.Type = D3D12_TEXTURE_COPY_TYPE_SUBRESOURCE_INDEX;
228 dstLoc.SubresourceIndex = 0;
229
230 cmdList->CopyTextureRegion(&dstLoc, 0, 0, 0, &srcLoc, nullptr);
231
232 // リソースバリア (コピー先 -> シェーダーリソース読み取り)
233 D3D12_RESOURCE_BARRIER barrier = {};
234 barrier.Type = D3D12_RESOURCE_BARRIER_TYPE_TRANSITION;
235 barrier.Transition.pResource = resource.Get();
236 barrier.Transition.StateBefore = D3D12_RESOURCE_STATE_COPY_DEST;
237 barrier.Transition.StateAfter = D3D12_RESOURCE_STATE_PIXEL_SHADER_RESOURCE;
238 barrier.Transition.Subresource = D3D12_RESOURCE_BARRIER_ALL_SUBRESOURCES;
239 cmdList->ResourceBarrier(1, &barrier);
240
241 cmdList->Close();
242
243 // 実行して待機
244 ID3D12CommandList* ppCommandLists[] = { cmdList.Get() };
245 commandQueue->ExecuteCommandLists(1, ppCommandLists);
246
247 // フェンス同期
248 ComPtr<ID3D12Fence> fence;
249 device->CreateFence(0, D3D12_FENCE_FLAG_NONE, IID_PPV_ARGS(&fence));
250 commandQueue->Signal(fence.Get(), 1);
251
252 HANDLE fenceEvent = CreateEvent(nullptr, FALSE, FALSE, nullptr);
253 if (fence->GetCompletedValue() < 1)
254 {
255 fence->SetEventOnCompletion(1, fenceEvent);
256 WaitForSingleObject(fenceEvent, INFINITE);
257 }
258 CloseHandle(fenceEvent);
259
260 return true;
261 }
