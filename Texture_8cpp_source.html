19 bool Texture::Initialize(ID3D12Device* device, ID3D12CommandQueue* commandQueue, const std::string& filepath)
20 {
21 SPAN_LOG("Loading Texture: %s", filepath.c_str());
22
23 m_FilePath = filepath;
24
25 // 1. stb_image で画像読み込み
26 int w, h, channels;
27 // RGBA (4チャンネル) で強制的に読み込む
28 unsigned char* data = stbi_load(filepath.c_str(), &w, &h, &channels, 4);
29
30 if (!data)
31 {
32 SPAN_ERROR("Failed to load image: %s", filepath.c_str());
33 return false;
34 }
35
36 width = static_cast<uint32_t>(w);
37 height = static_cast<uint32_t>(h);
38
39 // 2. GPUへアップロード
40 if (!UploadTexture(device, commandQueue, data, width, height, 4))
41 {
42 stbi_image_free(data);
43 return false;
44 }
45
46 // メモリ解放
47 stbi_image_free(data);
48
49 // 3. SRV (Shader Resource View) ヒープの作成
50 D3D12_DESCRIPTOR_HEAP_DESC srvHeapDesc = {};
51 srvHeapDesc.NumDescriptors = 1;
52 srvHeapDesc.Type = D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV;
53 srvHeapDesc.Flags = D3D12_DESCRIPTOR_HEAP_FLAG_SHADER_VISIBLE;
54
55 if (FAILED(device->CreateDescriptorHeap(&srvHeapDesc, IID_PPV_ARGS(&srvHeap))))
56 {
57 SPAN_ERROR("Failed to create SRV Heap");
58 return false;
59 }
60
61 // 4. SRV (ビュー) の作成
62 D3D12_SHADER_RESOURCE_VIEW_DESC srvDesc = {};
63 srvDesc.Shader4ComponentMapping = D3D12_DEFAULT_SHADER_4_COMPONENT_MAPPING;
64 srvDesc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
65 srvDesc.ViewDimension = D3D12_SRV_DIMENSION_TEXTURE2D;
66 srvDesc.Texture2D.MipLevels = 1;
67
68 srvHandleCPU = srvHeap->GetCPUDescriptorHandleForHeapStart();
69 device->CreateShaderResourceView(resource.Get(), &srvDesc, srvHandleCPU);
70
71 return true;
72 }

74 bool Texture::UploadTexture(ID3D12Device* device, ID3D12CommandQueue* commandQueue,
75 const void* initialData, uint64_t w, uint64_t h, uint64_t bytesPerPixel)
76 {
77 // リソース記述
78 D3D12_RESOURCE_DESC textureDesc = {};
79 textureDesc.MipLevels = 1;
80 textureDesc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
81 textureDesc.Width = w;
82 textureDesc.Height = h;
83 textureDesc.Flags = D3D12_RESOURCE_FLAG_NONE;
84 textureDesc.DepthOrArraySize = 1;
85 textureDesc.SampleDesc.Count = 1;
86 textureDesc.SampleDesc.Quality = 0;
87 textureDesc.Dimension = D3D12_RESOURCE_DIMENSION_TEXTURE2D;
88
89 // 1. デフォルトヒープ（VRAM）にテクスチャリソースを作成
90 // 初期状態は COPY_DEST (コピー先) にする
91 D3D12_HEAP_PROPERTIES defaultHeap = { D3D12_HEAP_TYPE_DEFAULT };
92
93 if (FAILED(device->CreateCommittedResource(
94 &defaultHeap,
95 D3D12_HEAP_FLAG_NONE,
96 &textureDesc,
97 D3D12_RESOURCE_STATE_COPY_DEST,
98 nullptr,
99 IID_PPV_ARGS(&resource))))
100 {
101 return false;
102 }
103
104 // 2. アップロード用バッファのサイズ計算
105 uint64_t uploadBufferSize = 0;
106 D3D12_PLACED_SUBRESOURCE_FOOTPRINT footprint;
107 UINT64 rowSizeInBytes;
108 UINT numRows;
109
110 device->GetCopyableFootprints(&textureDesc, 0, 1, 0, &footprint, &numRows, &rowSizeInBytes, &uploadBufferSize);
111
112 // 3. アップロードヒープ（CPU書き込み可）のリソース作成
113 D3D12_HEAP_PROPERTIES uploadHeap = { D3D12_HEAP_TYPE_UPLOAD };
114 D3D12_RESOURCE_DESC bufferDesc = {};
115 bufferDesc.Dimension = D3D12_RESOURCE_DIMENSION_BUFFER;
116 bufferDesc.Alignment = 0;
117 bufferDesc.Width = uploadBufferSize;
118 bufferDesc.Height = 1;
119 bufferDesc.DepthOrArraySize = 1;
120 bufferDesc.MipLevels = 1;
121 bufferDesc.Format = DXGI_FORMAT_UNKNOWN;
122 bufferDesc.SampleDesc.Count = 1;
123 bufferDesc.Layout = D3D12_TEXTURE_LAYOUT_ROW_MAJOR;
124 bufferDesc.Flags = D3D12_RESOURCE_FLAG_NONE;
125
126 if (FAILED(device->CreateCommittedResource(
127 &uploadHeap,
128 D3D12_HEAP_FLAG_NONE,
129 &bufferDesc,
130 D3D12_RESOURCE_STATE_GENERIC_READ,
131 nullptr,
132 IID_PPV_ARGS(&uploadBuffer))))
133 {
134 return false;
135 }
136
137 // 4. データをアップロードバッファにコピー
138 void* mappedData = nullptr;
139 uploadBuffer->Map(0, nullptr, &mappedData);
140
141 const uint8_t* srcData = reinterpret_cast<const uint8_t*>(initialData);
142 uint8_t* destData = reinterpret_cast<uint8_t*>(mappedData);
143
144 for (UINT i = 0; i < h; ++i)
145 {
146 memcpy(destData + (i * footprint.Footprint.RowPitch),
147 srcData + (i * w * bytesPerPixel),
148 w * bytesPerPixel);
149 }
150 uploadBuffer->Unmap(0, nullptr);
151
152 // 5. コマンドリストを作ってコピー命令を発行
153 // (本来はRendererのコマンドリストを使うべきだが、初期化用として一時的に作成)
154 ComPtr<ID3D12CommandAllocator> cmdAlloc;
155 ComPtr<ID3D12GraphicsCommandList> cmdList;
156 device->CreateCommandAllocator(D3D12_COMMAND_LIST_TYPE_DIRECT, IID_PPV_ARGS(&cmdAlloc));
157 device->CreateCommandList(0, D3D12_COMMAND_LIST_TYPE_DIRECT, cmdAlloc.Get(), nullptr, IID_PPV_ARGS(&cmdList));
158
159 D3D12_TEXTURE_COPY_LOCATION srcLoc = {};
160 srcLoc.pResource = uploadBuffer.Get();
161 srcLoc.Type = D3D12_TEXTURE_COPY_TYPE_PLACED_FOOTPRINT;
162 srcLoc.PlacedFootprint = footprint;
163
164 D3D12_TEXTURE_COPY_LOCATION dstLoc = {};
165 dstLoc.pResource = resource.Get();
166 dstLoc.Type = D3D12_TEXTURE_COPY_TYPE_SUBRESOURCE_INDEX;
167 dstLoc.SubresourceIndex = 0;
168
169 cmdList->CopyTextureRegion(&dstLoc, 0, 0, 0, &srcLoc, nullptr);
170
171 // リソースバリア (コピー先 -> シェーダーリソース読み取り)
172 D3D12_RESOURCE_BARRIER barrier = {};
173 barrier.Type = D3D12_RESOURCE_BARRIER_TYPE_TRANSITION;
174 barrier.Transition.pResource = resource.Get();
175 barrier.Transition.StateBefore = D3D12_RESOURCE_STATE_COPY_DEST;
176 barrier.Transition.StateAfter = D3D12_RESOURCE_STATE_PIXEL_SHADER_RESOURCE;
177 barrier.Transition.Subresource = D3D12_RESOURCE_BARRIER_ALL_SUBRESOURCES;
178 cmdList->ResourceBarrier(1, &barrier);
179
180 cmdList->Close();
181
182 // 実行して待機 (簡易同期)
183 ID3D12CommandList* ppCommandLists[] = { cmdList.Get() };
184 commandQueue->ExecuteCommandLists(1, ppCommandLists);
185
186 // フェンス同期 (待機)
187 ComPtr<ID3D12Fence> fence;
188 device->CreateFence(0, D3D12_FENCE_FLAG_NONE, IID_PPV_ARGS(&fence));
189 commandQueue->Signal(fence.Get(), 1);
190
191 HANDLE fenceEvent = CreateEvent(nullptr, FALSE, FALSE, nullptr);
192 if (fence->GetCompletedValue() < 1)
193 {
194 fence->SetEventOnCompletion(1, fenceEvent);
195 WaitForSingleObject(fenceEvent, INFINITE);
196 }
197 CloseHandle(fenceEvent);
198
199 return true;
200 }
