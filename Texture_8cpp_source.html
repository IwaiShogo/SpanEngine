17 bool Texture::Initialize(ID3D12Device* device, ID3D12CommandQueue* commandQueue, const std::string& filepath)
18 {
19 SPAN_LOG("Loading Texture: %s", filepath.c_str());

21
22 // 拡張子でHDRかどうかを判定
23 m_IsHDR = (filepath.length() > 4 && filepath.substr(filepath.length() - 4) == ".hdr");

33 data = stbi_loadf(filepath.c_str(), &w, &h, &channels, 4);
34 format = DXGI_FORMAT_R32G32B32A32_FLOAT;
35 bytesPerPixel = 16;
36 }
37 else
38 {
39 // LDR画像は unsigned char 配列として読み込む
40 data = stbi_load(filepath.c_str(), &w, &h, &channels, 4);

42

45 SPAN_ERROR("Failed to load image: %s", filepath.c_str());

47 }
48

50 height = static_cast<uint32_t>(h);
51
52 // 2. GPUへアップロード

54 {
55 stbi_image_free(data);
56 return false;
57 }

59 // メモリ解放
60 stbi_image_free(data);
61
62 // 3. SRV (Shader Resource View) ヒープの作成
63 D3D12_DESCRIPTOR_HEAP_DESC srvHeapDesc = {};
64 srvHeapDesc.NumDescriptors = 1;

66 srvHeapDesc.Flags = D3D12_DESCRIPTOR_HEAP_FLAG_NONE;
67

69 {
70 SPAN_ERROR("Failed to create SRV Heap");
71 return false;
72 }
73
74 // 4. SRV (ビュー) の作成
75 D3D12_SHADER_RESOURCE_VIEW_DESC srvDesc = {};
76 srvDesc.Shader4ComponentMapping = D3D12_DEFAULT_SHADER_4_COMPONENT_MAPPING;

79 srvDesc.Texture2D.MipLevels = 1;

232 bool Texture::UploadTexture(ID3D12Device* device, ID3D12CommandQueue* commandQueue,
233 const void* initialData, uint64_t w, uint64_t h, uint64_t bytesPerPixel, DXGI_FORMAT format)
234 {
235 // リソース記述
236 D3D12_RESOURCE_DESC textureDesc = {};
237 textureDesc.MipLevels = 1;
238 textureDesc.Format = format;
239 textureDesc.Width = w;
240 textureDesc.Height = static_cast<UINT>(h);
241 textureDesc.Flags = D3D12_RESOURCE_FLAG_NONE;
242 textureDesc.DepthOrArraySize = 1;
243 textureDesc.SampleDesc.Count = 1;
244 textureDesc.SampleDesc.Quality = 0;
245 textureDesc.Dimension = D3D12_RESOURCE_DIMENSION_TEXTURE2D;
246
247 // 1. デフォルトヒープ（VRAM）にテクスチャリソースを作成
248 D3D12_HEAP_PROPERTIES defaultHeap = { D3D12_HEAP_TYPE_DEFAULT };
249
250 if (FAILED(device->CreateCommittedResource(
251 &defaultHeap,
252 D3D12_HEAP_FLAG_NONE,
253 &textureDesc,
254 D3D12_RESOURCE_STATE_COPY_DEST,
255 nullptr,
256 IID_PPV_ARGS(&resource))))
257 {
258 return false;
259 }
260
261 // 2. アップロード用バッファのサイズ計算
262 uint64_t uploadBufferSize = 0;
263 D3D12_PLACED_SUBRESOURCE_FOOTPRINT footprint;
264 UINT64 rowSizeInBytes;
265 UINT numRows;
266 device->GetCopyableFootprints(&textureDesc, 0, 1, 0, &footprint, &numRows, &rowSizeInBytes, &uploadBufferSize);
267
268 // 3. アップロードヒープ（CPU書き込み可）のリソース作成
269 D3D12_HEAP_PROPERTIES uploadHeap = { D3D12_HEAP_TYPE_UPLOAD };
270 D3D12_RESOURCE_DESC bufferDesc = {};
271 bufferDesc.Dimension = D3D12_RESOURCE_DIMENSION_BUFFER;
272 bufferDesc.Alignment = 0;
273 bufferDesc.Width = uploadBufferSize;
274 bufferDesc.Height = 1;
275 bufferDesc.DepthOrArraySize = 1;
276 bufferDesc.MipLevels = 1;
277 bufferDesc.Format = DXGI_FORMAT_UNKNOWN;
278 bufferDesc.SampleDesc.Count = 1;
279 bufferDesc.Layout = D3D12_TEXTURE_LAYOUT_ROW_MAJOR;
280 bufferDesc.Flags = D3D12_RESOURCE_FLAG_NONE;
281
282 if (FAILED(device->CreateCommittedResource(
283 &uploadHeap,
284 D3D12_HEAP_FLAG_NONE,
285 &bufferDesc,
286 D3D12_RESOURCE_STATE_GENERIC_READ,
287 nullptr,
288 IID_PPV_ARGS(&uploadBuffer))))
289 {
290 return false;
291 }
292
293 // 4. データをアップロードバッファにコピー
294 void* mappedData = nullptr;
295 uploadBuffer->Map(0, nullptr, &mappedData);
296
297 const uint8_t* srcData = reinterpret_cast<const uint8_t*>(initialData);
298 uint8_t* destData = reinterpret_cast<uint8_t*>(mappedData);
299
300 for (UINT i = 0; i < h; ++i)
301 {
302 memcpy(destData + (i * footprint.Footprint.RowPitch),
303 srcData + (i * w * bytesPerPixel),
304 w * bytesPerPixel);
305 }
306 uploadBuffer->Unmap(0, nullptr);
307
308 // 5. コマンドリストを作ってコピー命令を発行
309 ComPtr<ID3D12CommandAllocator> cmdAlloc;
310 ComPtr<ID3D12GraphicsCommandList> cmdList;
311 device->CreateCommandAllocator(D3D12_COMMAND_LIST_TYPE_DIRECT, IID_PPV_ARGS(&cmdAlloc));
312 device->CreateCommandList(0, D3D12_COMMAND_LIST_TYPE_DIRECT, cmdAlloc.Get(), nullptr, IID_PPV_ARGS(&cmdList));
313
314 D3D12_TEXTURE_COPY_LOCATION srcLoc = {};
315 srcLoc.pResource = uploadBuffer.Get();
316 srcLoc.Type = D3D12_TEXTURE_COPY_TYPE_PLACED_FOOTPRINT;
317 srcLoc.PlacedFootprint = footprint;
318
319 D3D12_TEXTURE_COPY_LOCATION dstLoc = {};
320 dstLoc.pResource = resource.Get();
321 dstLoc.Type = D3D12_TEXTURE_COPY_TYPE_SUBRESOURCE_INDEX;
322 dstLoc.SubresourceIndex = 0;
323
324 cmdList->CopyTextureRegion(&dstLoc, 0, 0, 0, &srcLoc, nullptr);
325
326 // リソースバリア (コピー先 -> シェーダーリソース読み取り)
327 D3D12_RESOURCE_BARRIER barrier = {};
328 barrier.Type = D3D12_RESOURCE_BARRIER_TYPE_TRANSITION;
329 barrier.Transition.pResource = resource.Get();
330 barrier.Transition.StateBefore = D3D12_RESOURCE_STATE_COPY_DEST;
331 barrier.Transition.StateAfter = D3D12_RESOURCE_STATE_PIXEL_SHADER_RESOURCE;
332 barrier.Transition.Subresource = D3D12_RESOURCE_BARRIER_ALL_SUBRESOURCES;
333 cmdList->ResourceBarrier(1, &barrier);
334
335 cmdList->Close();
336
337 // 実行して待機
338 ID3D12CommandList* ppCommandLists[] = { cmdList.Get() };
339 commandQueue->ExecuteCommandLists(1, ppCommandLists);
340
341 // フェンス同期
342 ComPtr<ID3D12Fence> fence;
343 device->CreateFence(0, D3D12_FENCE_FLAG_NONE, IID_PPV_ARGS(&fence));
344 commandQueue->Signal(fence.Get(), 1);
345
346 HANDLE fenceEvent = CreateEvent(nullptr, FALSE, FALSE, nullptr);
347 if (fenceEvent)
348 {
349 if (fence->GetCompletedValue() < 1)
350 {
351 fence->SetEventOnCompletion(1, fenceEvent);
352 WaitForSingleObject(fenceEvent, INFINITE);
353 }
354 CloseHandle(fenceEvent);
355 }
356
357 return true;
358 }
