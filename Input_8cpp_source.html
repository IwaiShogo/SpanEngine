60 void Input::Update()
61 {
62 // ロック中は「移動量」を計算した後、カーソルを中央に戻す
63 if (isCursorLocked && hWnd)
64 {
65 // 1. 現在のカーソル位置（スクリーン座標）を取得
66 POINT curP;
67 GetCursorPos(&curP);
68
69 // 2. ウィンドウ中央（スクリーン座標）を計算
70 RECT rect;
71 GetClientRect(hWnd, &rect);
72 POINT center = { (rect.right - rect.left) / 2, (rect.bottom - rect.top) / 2 };
73 ClientToScreen(hWnd, &center);
74
75 // 3. 中央からのズレを移動量(Delta)とする
76 // これにより、前回位置に関係なく純粋な移動量が取れる
77 if (ignoreNextDelta)
78 {
79 mouseDelta = { 0, 0 };
80 ignoreNextDelta = false;
81 }
82 else
83 {
84 mouseDelta.x = (float)(curP.x - center.x);
85 mouseDelta.y = (float)(curP.y - center.y);
86 }
87
88 // 4. カーソルを物理的に中央に戻す
89 SetCursorPos(center.x, center.y);
90
91 // 5. 内部座標も中央固定にしておく
92 ScreenToClient(hWnd, &center);
93 mousePosition = { (float)center.x, (float)center.y };
94 prevMousePosition = mousePosition;
95 }
96 else
97 {
98 // 通常
99 mouseDelta = mousePosition - prevMousePosition;
100 prevMousePosition = mousePosition;
101 }
102
103 // 2. コントローラーの前回状態保存
104 memcpy(prevGamepadStates, gamepadStates, sizeof(bool) * 20);
105
106 // 3. XInputポーリング (コントローラー0番のみ対応)
107 XINPUT_STATE state;
108 ZeroMemory(&state, sizeof(XINPUT_STATE));
109
110 // 接続確認と取得
111 if (XInputGetState(0, &state) == ERROR_SUCCESS)
112 {
113 isConnected = true;
114
115 // --- ボタンの読み取り ---
116 auto CheckBtn = [&](int idx, int mask) { gamepadStates[idx] = (state.Gamepad.wButtons & mask) != 0; };
117 CheckBtn(0, XINPUT_GAMEPAD_A);
118 CheckBtn(1, XINPUT_GAMEPAD_B);
119 CheckBtn(2, XINPUT_GAMEPAD_X);
120 CheckBtn(3, XINPUT_GAMEPAD_Y);
121 CheckBtn(4, XINPUT_GAMEPAD_START);
122 CheckBtn(5, XINPUT_GAMEPAD_BACK);
123 CheckBtn(6, XINPUT_GAMEPAD_LEFT_SHOULDER);
124 CheckBtn(7, XINPUT_GAMEPAD_RIGHT_SHOULDER);
125 CheckBtn(8, XINPUT_GAMEPAD_DPAD_UP);
126 CheckBtn(9, XINPUT_GAMEPAD_DPAD_DOWN);
127 CheckBtn(10, XINPUT_GAMEPAD_DPAD_LEFT);
128 CheckBtn(11, XINPUT_GAMEPAD_DPAD_RIGHT);
129
130 // --- スティックの読み取りとデッドゾーン処理 ---
131 auto ApplyDeadzone = [](short val) -> float {
132 if (std::abs(val) < XINPUT_GAMEPAD_LEFT_THUMB_DEADZONE) return 0.0f;
133 return val / 32767.0f;
134 };
135
136 gamepadAxes[(int)Axis::LeftStickX] = ApplyDeadzone(state.Gamepad.sThumbLX);
137 gamepadAxes[(int)Axis::LeftStickY] = ApplyDeadzone(state.Gamepad.sThumbLY);
138 gamepadAxes[(int)Axis::RightStickX] = ApplyDeadzone(state.Gamepad.sThumbRX);
139 gamepadAxes[(int)Axis::RightStickY] = ApplyDeadzone(state.Gamepad.sThumbRY);
140
141 // トリガー (0-255 -> 0.0-1.0)
142 gamepadAxes[(int)Axis::LeftTrigger] = state.Gamepad.bLeftTrigger / 255.0f;
143 gamepadAxes[(int)Axis::RightTrigger] = state.Gamepad.bRightTrigger / 255.0f;
144 }
145 else
146 {
147 isConnected = false;
148 ZeroMemory(gamepadStates, sizeof(bool) * 20);
149 ZeroMemory(gamepadAxes, sizeof(float) * 6);
150 }
151 }
