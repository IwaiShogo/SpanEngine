68 // --- Directional Light ---

72 world->ForEach<DirectionalLight, LocalToWorld>(
73 [&](Entity, DirectionalLight& dl, LocalToWorld& ltw)
74 {
75 LightDataGPU ld = {};
76 ld.Type = 0;
77 ld.Direction = Vector3::Normalize(Vector3(ltw.Value.m[2][0], ltw.Value.m[2][1], ltw.Value.m[2][2]));
78 ld.Color = dl.Color; ld.Intensity = dl.Intensity; ld.CastShadows = dl.CastShadows ? 1 : 0;

81 Vector3 lightTarget = renderer.GetCameraPosition();
82 Vector3 lightPos = lightTarget - ld.Direction * (dl.ShadowMaxDistance * 0.5f);
83 Vector3 up = (std::abs(Vector3::Dot(ld.Direction, Vector3::Up)) > 0.999f) ? Vector3::Forward : Vector3::Up;
84 dirLightMatrix = Matrix4x4::LookAtLH(lightPos, lightTarget, up) * Matrix4x4::OrthographicLH(dl.ShadowAreaSize, dl.ShadowAreaSize, 1.0f, dl.ShadowMaxDistance);

86 ld.ShadowMatrix = dirLightMatrix;

92 // --- Spot Light ---
93 int spotShadowCount = 0;
94
95 world->ForEach<SpotLight, LocalToWorld>(
96 [&](Entity, SpotLight& sl, LocalToWorld& ltw)
97 {
98 LightDataGPU ld = {};
99 ld.Type = 2;
100 ld.Position = Vector3(ltw.Value.m[3][0], ltw.Value.m[3][1], ltw.Value.m[3][2]);
101 ld.Direction = Vector3::Normalize(Vector3(ltw.Value.m[2][0], ltw.Value.m[2][1], ltw.Value.m[2][2]));
102 ld.Color = sl.Color; ld.Intensity = sl.Intensity; ld.Range = sl.Range;
103 ld.InnerConeAngle = std::cos(Deg2Rad(sl.InnerConeAngle)); ld.OuterConeAngle = std::cos(Deg2Rad(sl.OuterConeAngle));
104 ld.CastShadows = sl.CastShadows ? 1 : 0;
105 ld.ShadowIndex = -1;
106
107 // ÂΩ±„ÇíËêΩ„Å®„ÅôË®≠ÂÆö„Åß„ÄÅ„Åã„Å§ÊúÄÂ§ßÊï∞(4)Êú™Ê∫Ä„Å™„ÇâË°åÂàó„ÇíË®àÁÆó
108 if (sl.CastShadows && spotShadowCount < 4)
109 {
110 ld.ShadowIndex = spotShadowCount;
111 Vector3 up = (std::abs(Vector3::Dot(ld.Direction, Vector3::Up)) > 0.999f) ? Vector3::Forward : Vector3::Up;
112 Matrix4x4 view = Matrix4x4::LookAtLH(ld.Position, ld.Position + ld.Direction, up);
113 // ÈÄèË¶ñÊäïÂΩ±(Perspective)„Çí‰ΩøÁî®ÔºÅ Fov „ÅØ OuterCone „ÅÆ2ÂÄç„ÅÆÂ∫É„Åï

115 ld.ShadowMatrix = view * proj;
116 spotShadowCount++;
117 }
118 else
119 {
120 ld.ShadowMatrix = Matrix4x4::Identity().Transpose();
121 }
122
123 activeLights.push_back(ld);
124 }
125 );
126

128 int pointShadowCount = 0;
129

137 ld.Intensity = pl.Intensity;
138 ld.Range = pl.Range;
139 ld.CastShadows = pl.CastShadows ? 1 : 0;
140 ld.ShadowIndex = -1;
141
142 if (pl.CastShadows && pointShadowCount < 1)
143 {
144 ld.ShadowIndex = pointShadowCount;
145 pointShadowCount++;
146 }
147 activeLights.push_back(ld);
148 }
149 );

223 }

243 // 6ÊñπÂêë„ÅÆ„Ç´„É°„É©Ë®≠ÂÆö
244 Matrix4x4 views[6] = {

259 pointPass->SetRenderTarget(cmd, face);

261

263 if (mf.mesh && mr.material && mr.CastShadows)
264 pointPass->DrawMesh(&renderer, cmd, mf.mesh, ltw.Value, cubeMatrix);
265 });
266 }
267 }
268 }
269 pointPass->EndPass(cmd);

328 // [2] „Ç¨„É©„Çπ (Opaque „Å†„Åå Transmission „Åå 0 „Çà„ÇäÂ§ß„Åç„ÅÑ„ÇÇ„ÅÆ

330 [&](Entity, MeshFilter& mf, MeshRenderer& mr, LocalToWorld& ltw)
331 {
332 if (!mf.mesh || !mr.material) return;
333
334 if (mr.material->GetBlendMode() != BlendMode::Transparent && mr.material->GetData().Transmission > 0.0f)
335 {
336 renderer.DrawMesh(mf.mesh, mr.material, ltw.Value);
337 }

350 }

353 }
