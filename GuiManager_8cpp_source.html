163 D3D12_GPU_DESCRIPTOR_HANDLE GuiManager::RegisterTexture(D3D12_CPU_DESCRIPTOR_HANDLE srcHandle)
164 {
165 if (!m_device || !srvHeap || srcHandle.ptr == 0) return { 0 };
166
167 // SRVヒープの先頭ハンドルを取得
168 D3D12_CPU_DESCRIPTOR_HANDLE destHandleCPU = srvHeap->GetCPUDescriptorHandleForHeapStart();
169 D3D12_GPU_DESCRIPTOR_HANDLE destHandleGPU = srvHeap->GetGPUDescriptorHandleForHeapStart();
170
171 // インクリメントサイズを取得
172 UINT handleIncrementSize = m_device->GetDescriptorHandleIncrementSize(D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV);
173
174 // インデックス1番目を使用する (0番目はImGuiのフォント用に使われているため)
175 int index = 1;
176
177 destHandleCPU.ptr += (UINT64)index * handleIncrementSize;
178 destHandleGPU.ptr += (UINT64)index * handleIncrementSize;
179
180 // デバイスを使ってコピーを実行
181 m_device->CopyDescriptorsSimple(1, destHandleCPU, srcHandle, D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV);
182
183 return destHandleGPU;
184 }
