23 AssetMetadata Span::AssetSerializer::LoadOrCreateMetadata(const std::filesystem::path& assetPath)
24 {
25 std::filesystem::path metaPath = assetPath;
26 metaPath += ".meta";
27
28 AssetMetadata metadata;
29
30 // 1. 既存の.metaを読み込む
31 if (std::filesystem::exists(metaPath))
32 {
33 std::ifstream stream(metaPath);
34 if (stream.is_open())
35 {
36 std::string line;
37 while (std::getline(stream, line))
38 {
39 if (line.find("GUID:") != std::string::npos)
40 {
41 try
42 {
43 metadata.Handle = std::stoull(line.substr(6));
44 }
45 catch (...) { metadata.Handle = 0; }
46 }
47 else if (line.find("Type:") != std::string::npos)
48 {
49 try
50 {
51 metadata.Type = (AssetType)std::stoi(line.substr(6));
52 }
53 catch (...) { metadata.Type = AssetType::None; }
54 }
55 }
56 stream.close();
57 }
58 }
59
60 // 2. 読み込み失敗、または存在しない場合は新規作成
61 if (!metadata.IsValid())
62 {
63 metadata.Handle = s_UniformDistribution(s_Engine);
64 if (metadata.Handle == 0) metadata.Handle++;
65
66 metadata.Type = DeduceTypeFromExtension(assetPath);
67
68 SaveMetadata(assetPath, metadata);
69 SPAN_LOG("Generated .meta for: %s", assetPath.string().c_str());
70 }
71
72 return metadata;
73 }

89 AssetType AssetSerializer::DeduceTypeFromExtension(const std::filesystem::path& path)
90 {
91 std::string ext = path.extension().string();
92 if (ext == ".png" || ext == ".jpg" || ext == ".tga") return AssetType::Texture;
93 if (ext == ".fbx" || ext == ".obj") return AssetType::Mesh;
94 if (ext == ".mat") return AssetType::Material;
95 if (ext == ".span") return AssetType::Scene;
96 if (ext == ".h" || ext == ".cpp") return AssetType::Script;
97 return AssetType::None;
98 }
