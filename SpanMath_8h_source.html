303 inline Vector3 Quaternion::ToEuler() const
304 {
305 // Matrixを経由して正確にEuler角を取り出す
306 Matrix4x4 m = Matrix4x4::Rotation(*this);
307
308 float pitch, yaw, roll;
309
310 // Pitch (X軸回転)
311 if (m._32 < -0.999f) pitch = HalfPI;
312 else if (m._32 > 0.999f) pitch = -HalfPI;
313 else pitch = asin(-m._32);
314
315 // Yaw (Y軸回転) & Roll (Z軸回転)
316 if (abs(m._32) < 0.999f)
317 {
318 yaw = atan2(m._31, m._33);
319 roll = atan2(m._12, m._22);
320 }
321 else
322 {
323 yaw = atan2(-m._13, m._11);
324 roll = 0.0f;
325 }
326
327 return Vector3(pitch, yaw, roll);
328 }
