24 inline std::filesystem::path GetTrashPath(const std::filesystem::path& originalPath)
25 {
26 auto projectRoot = originalPath.parent_path();
27 while (projectRoot.has_parent_path() && projectRoot.filename() != "Assets")
28 {
29 projectRoot = projectRoot.parent_path();
30 }
31 if (projectRoot.filename() == "Assets") projectRoot = projectRoot.parent_path();
32
33 std::filesystem::path trashDir = projectRoot / ".Trash";
34
35 std::error_code ec;
36 if (!std::filesystem::exists(trashDir)) std::filesystem::create_directories(trashDir, ec);
37
38 // 一意な名前を生成
39 auto now = std::chrono::system_clock::now().time_since_epoch().count();
40 std::string filename = originalPath.stem().string();
41 std::string ext = originalPath.extension().string();
42
43 std::stringstream ss;
44 ss << filename << "_" << now << ext;
45
46 return trashDir / ss.str();
47 }

111
112 private:

114 std::filesystem::path m_DestinationDir;
115 std::filesystem::path m_DestinationPath; // 移動後のフルパス
116 };

118 /**
119 * @class DeleteFileCommand
120 * @brief ファイルの削除コマンド。
121 */
122 class DeleteFileCommand : public ICommand
123 {
124 public:
125 DeleteFileCommand(const std::filesystem::path& path)
126 : m_OriginalPath(path)
127 {
128 m_TrashPath = GetTrashPath(path);
129 }
130
131 bool Execute() override
132 {
133 try
134 {
135 std::filesystem::rename(m_OriginalPath, m_TrashPath);
136
137 // .metaも削除
138 auto srcMeta = m_OriginalPath.string() + ".meta";
139 if (std::filesystem::exists(srcMeta))
140 {
141 std::filesystem::rename(srcMeta, m_TrashPath.string() + ".meta");
142 }
143 return true;
144 }
145 catch (...) { return false; }
146 }
147
148 void Undo() override
149 {
150 try
151 {
152 // 元のディレクトリが消えている場合に備えて再生成
153 if (!std::filesystem::exists(m_OriginalPath.parent_path()))
154 {
155 std::filesystem::create_directories(m_OriginalPath.parent_path());
156 }
157
158 std::filesystem::rename(m_TrashPath, m_OriginalPath);
159
160 auto trashMeta = m_TrashPath.string() + ".meta";
161 if (std::filesystem::exists(trashMeta))
162 {
163 std::filesystem::rename(trashMeta, m_OriginalPath.string() + ".meta");
164 }
165 }
166 catch (...) {}
167 }
168
169 const char* GetName() const override { return "Delete Asset"; }
170
171 private:
172 std::filesystem::path m_OriginalPath;
173 std::filesystem::path m_TrashPath;
174 };
