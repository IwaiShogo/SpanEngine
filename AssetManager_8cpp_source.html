35 void AssetManager::Shutdown()
36 {
37 std::lock_guard<std::recursive_mutex> lock(m_Mutex);
38
39 // 全テクスチャの明示的な解放
40 for (auto& [key, texture] : m_TextureCache)
41 {
42 if (texture) texture->Shutdown();
43 }
44 m_TextureCache.clear();
45 m_MeshCache.clear();
46 m_MaterialCache.clear();
47
48 m_DefaultMaterial.reset();
49 m_DefaultVS.reset();
50 m_DefaultPS.reset();
51
52 AssetRegistry::Get().Shutdown();
53
54 SPAN_LOG("AssetManager Shutdown: All assets released.");
55 }

57 std::shared_ptr<Texture> AssetManager::GetTexture(AssetHandle handle)
58 {
59 if (handle == 0) return nullptr;
60
61 std::lock_guard<std::recursive_mutex> lock(m_Mutex);
62
63 // 1. キャッシュ確認
64 auto it = m_TextureCache.find(handle);
65 if (it != m_TextureCache.end()) return it->second;
66
67 // 2. パス解決
68 const auto& path = AssetRegistry::Get().GetPath(handle);
69 if (path.empty())
70 {
71 SPAN_WARN("AssetManager: GUID %llu not found in Registry.", handle);
72 return nullptr;
73 }
74
75 // 3. ロード
76 // 将来的にはここで AsyncLoad
77 auto texture = std::make_shared<Texture>();
78 if (texture->Initialize(m_Device, m_CommandQueue, path.string()))
79 {
80 texture->Handle = handle;
81 m_TextureCache[handle] = texture;
82 return texture;
83 }
84
85 return nullptr;
86 }

88 std::shared_ptr<Mesh> AssetManager::GetMesh(AssetHandle handle)
89 {
90 if (handle == 0) return nullptr;
91 std::lock_guard<std::recursive_mutex> lock(m_Mutex);
92
93 if (m_MeshCache.find(handle) != m_MeshCache.end()) return m_MeshCache[handle];
94
95 const auto& path = AssetRegistry::Get().GetPath(handle);
96 if (path.empty()) return nullptr;
97
98 auto meshes = ModelLoader::Load(m_Device, path.string());
99 if (!meshes.empty() && meshes[0] != nullptr)
100 {
101 m_MeshCache[handle] = std::shared_ptr<Mesh>(meshes[0]);
102 m_MeshCache[handle]->SetPath(path.string());
103
104 // 使用しなかった残りのメッシュを削除
105 for (size_t i = 1; i < meshes.size(); ++i)
106 {
107 delete meshes[i];
108 }
109
110 return m_MeshCache[handle];
111 }
112 return nullptr;
113 }

115 std::shared_ptr<Material> AssetManager::GetMaterial(AssetHandle handle)
116 {
117 if (handle == 0) return nullptr;
118 std::lock_guard<std::recursive_mutex> lock(m_Mutex);
119
120 // 1. キャッシュ確認
121 auto it = m_MaterialCache.find(handle);
122 if (it != m_MaterialCache.end()) return it->second;
123
124 // 2. パス解決
125 const auto& path = AssetRegistry::Get().GetPath(handle);
126 if (path.empty()) return nullptr;
127
128 // 3. ロード
129 auto material = std::make_shared<Material>();
130 if (material->Deserialize(path))
131 {
132 material->Handle = handle;
133
134 if (material->AlbedoMapGUID != 0) material->SetAlbedoMap(GetTexture(material->AlbedoMapGUID).get());
135 if (material->NormalMapGUID != 0) material->SetNormalMap(GetTexture(material->NormalMapGUID).get());
136 if (material->MetallicMapGUID != 0) material->SetMetallicMap(GetTexture(material->MetallicMapGUID).get());
137 if (material->RoughnessMapGUID != 0) material->SetRoughnessMap(GetTexture(material->RoughnessMapGUID).get());
138 if (material->AOMapGUID != 0) material->SetAOMap(GetTexture(material->AOMapGUID).get());
139 if (material->EmissiveMapGUID != 0) material->SetEmissiveMap(GetTexture(material->EmissiveMapGUID).get());
140
141 // デフォルトシェーダーの割り当て
142 auto defaultMat = GetDefaultMaterial();
143 if (defaultMat)
144 {
145 material->SetShaders(defaultMat->GetVertexShader(), defaultMat->GetPixelShader());
146 }
147
148 material->Initialize(m_Device);
149 m_MaterialCache[handle] = material;
150 return material;
151 }
152 }

209 std::shared_ptr<Material> AssetManager::GetDefaultMaterial()
210 {
211 std::lock_guard<std::recursive_mutex> lock(m_Mutex);
212
213 if (m_DefaultMaterial)
214 {
215 return m_DefaultMaterial;
216 }
217
218 if (!m_Device) return nullptr;
219
220 if (!m_DefaultVS)
221 {
222 m_DefaultVS = std::make_shared<Shader>();
223 if (!m_DefaultVS->Load(L"Basic.hlsl", ShaderType::Vertex, "VSMain"))
224 {
225 SPAN_ERROR("Failed to load Default Vertex Shader");
226 }
227 }
228
229 if (!m_DefaultPS)
230 {
231 m_DefaultPS = std::make_shared<Shader>();
232 if (!m_DefaultPS->Load(L"Basic.hlsl", ShaderType::Pixel, "PSMain"))
233 {
234 SPAN_ERROR("Failed to load Default Pixel Shader");
235 }
236 }
237
238 // 2. マテリアルの生成
239 m_DefaultMaterial = std::make_shared<Material>();
240
241 if (m_DefaultMaterial->Initialize(m_Device))
242 {
243 m_DefaultMaterial->SetShaders(m_DefaultVS.get(), m_DefaultPS.get());
244
245 // 色などを設定
246 m_DefaultMaterial->GetData().AlbedoColor = Vector4(1.0f, 1.0f, 1.0f, 1.0f);
247 m_DefaultMaterial->Update();
248
249 SPAN_LOG("Default Material Created.");
250 }
251 else
252 {
253 SPAN_ERROR("Failed to initialize Default Material");
254 m_DefaultMaterial.reset();
255 }
256
257 return m_DefaultMaterial;
258 }
