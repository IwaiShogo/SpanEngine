16 bool IBLBuilder::Initialize(ID3D12Device* device)
17 {
18 // 1. Compute Shader のロード
19 m_equirectToCubemapCS = new Shader();
20 if (!m_equirectToCubemapCS->Load(L"EquirectangularToCubemap.hlsl", ShaderType::Compute, "CSMain"))
21 {
22 SPAN_ERROR("Failed to load EquirectangularToCubemap Compute Shader.");
23 return false;
24 }
25
26 // Irradiance Map用 Compute Shader のロード
27 m_irradianceCS = new Shader();
28 if (!m_irradianceCS->Load(L"IrradianceMap.hlsl", ShaderType::Compute, "CSMain"))
29 {
30 SPAN_ERROR("Failed to load IrradianceMap Compute Shader.");
31 return false;
32 }
33
34 // Prefilter Map用 Compute shader のロード
35 m_prefilterCS = new Shader();
36 if (!m_prefilterCS->Load(L"PrefilterMap.hlsl", ShaderType::Compute, "CSMain"))
37 {
38 SPAN_ERROR("Failed to load PrefilterMap Compute Shader.");
39 return false;
40 }
41
42 // BRDF Integration LUT用 Compute Shader のロード
43 m_brdfCS = new Shader();
44 if (!m_brdfCS->Load(L"BRDFIntegration.hlsl", ShaderType::Compute, "CSMain"))
45 {
46 SPAN_ERROR("Failed to load BRDFIntegration Compute Shader.");
47 return false;
48 }
49
50 // 2. Compute Root Signature の作成
51 D3D12_DESCRIPTOR_RANGE srvRange = {};
52 srvRange.RangeType = D3D12_DESCRIPTOR_RANGE_TYPE_SRV;
53 srvRange.NumDescriptors = 1;
54 srvRange.BaseShaderRegister = 0;
55 srvRange.RegisterSpace = 0;
56 srvRange.OffsetInDescriptorsFromTableStart = D3D12_DESCRIPTOR_RANGE_OFFSET_APPEND;
57
58 D3D12_DESCRIPTOR_RANGE uavRange = {};
59 uavRange.RangeType = D3D12_DESCRIPTOR_RANGE_TYPE_UAV;
60 uavRange.NumDescriptors = 1;
61 uavRange.BaseShaderRegister = 0;
62 uavRange.RegisterSpace = 0;
63 uavRange.OffsetInDescriptorsFromTableStart = D3D12_DESCRIPTOR_RANGE_OFFSET_APPEND;
64
65 D3D12_ROOT_PARAMETER rootParameters[3];
66 rootParameters[0].ParameterType = D3D12_ROOT_PARAMETER_TYPE_DESCRIPTOR_TABLE;
67 rootParameters[0].DescriptorTable.NumDescriptorRanges = 1;
68 rootParameters[0].DescriptorTable.pDescriptorRanges = &srvRange;
69 rootParameters[0].ShaderVisibility = D3D12_SHADER_VISIBILITY_ALL;
70
71 rootParameters[1].ParameterType = D3D12_ROOT_PARAMETER_TYPE_DESCRIPTOR_TABLE;
72 rootParameters[1].DescriptorTable.NumDescriptorRanges = 1;
73 rootParameters[1].DescriptorTable.pDescriptorRanges = &uavRange;
74 rootParameters[1].ShaderVisibility = D3D12_SHADER_VISIBILITY_ALL;
75
76 rootParameters[2].ParameterType = D3D12_ROOT_PARAMETER_TYPE_32BIT_CONSTANTS;
77 rootParameters[2].Constants.ShaderRegister = 0;
78 rootParameters[2].Constants.RegisterSpace = 0;
79 rootParameters[2].Constants.Num32BitValues = 2;
80 rootParameters[2].ShaderVisibility = D3D12_SHADER_VISIBILITY_ALL;
81
82 // サンプラー
83 D3D12_STATIC_SAMPLER_DESC sampler = {};
84 sampler.Filter = D3D12_FILTER_MIN_MAG_MIP_LINEAR;
85 sampler.AddressU = D3D12_TEXTURE_ADDRESS_MODE_WRAP;
86 sampler.AddressV = D3D12_TEXTURE_ADDRESS_MODE_CLAMP;
87 sampler.AddressW = D3D12_TEXTURE_ADDRESS_MODE_CLAMP;
88 sampler.MipLODBias = 0;
89 sampler.MaxAnisotropy = 1;
90 sampler.ComparisonFunc = D3D12_COMPARISON_FUNC_NEVER;
91 sampler.MinLOD = 0.0f;
92 sampler.MaxLOD = D3D12_FLOAT32_MAX;
93 sampler.ShaderRegister = 0; // s0
94 sampler.RegisterSpace = 0;
95 sampler.ShaderVisibility = D3D12_SHADER_VISIBILITY_ALL;
96
97 D3D12_ROOT_SIGNATURE_DESC rootSigDesc = {};
98 rootSigDesc.NumParameters = _countof(rootParameters);
99 rootSigDesc.pParameters = rootParameters;
100 rootSigDesc.NumStaticSamplers = 1;
101 rootSigDesc.pStaticSamplers = &sampler;
102 rootSigDesc.Flags = D3D12_ROOT_SIGNATURE_FLAG_NONE;
103
104 ComPtr<ID3DBlob> signature, error;
105 if (FAILED(D3D12SerializeRootSignature(&rootSigDesc, D3D_ROOT_SIGNATURE_VERSION_1, &signature, &error)))
106 {
107 SPAN_ERROR("IBLBuilder RootSignature Serialize Failed: %s", (char*)error->GetBufferPointer());
108 return false;
109 }
110
111 if (FAILED(device->CreateRootSignature(0, signature->GetBufferPointer(), signature->GetBufferSize(), IID_PPV_ARGS(&m_computeRootSignature))))
112 return false;
113
114 // 3. Compute Pipeline State Object (PSO) の作成
115 D3D12_COMPUTE_PIPELINE_STATE_DESC psoDesc = {};
116 psoDesc.pRootSignature = m_computeRootSignature.Get();
117 psoDesc.CS = m_equirectToCubemapCS->GetBytecode();
118
119 if (FAILED(device->CreateComputePipelineState(&psoDesc, IID_PPV_ARGS(&m_equirectToCubemapPSO))))
120 {
121 SPAN_ERROR("Failed to create IBLBuilder Compute PSO.");
122 return false;
123 }
124
125 // Irradiance Map用 Pipeline State Object の作成
126 D3D12_COMPUTE_PIPELINE_STATE_DESC irrPsoDesc = {};
127 irrPsoDesc.pRootSignature = m_computeRootSignature.Get();
128 irrPsoDesc.CS = m_irradianceCS->GetBytecode();
129
130 if (FAILED(device->CreateComputePipelineState(&irrPsoDesc, IID_PPV_ARGS(&m_irradiancePSO))))
131 {
132 SPAN_ERROR("Failed to create Irradiance Map Compute PSO.");
133 return false;
134 }
135
136 // Prefilter Map用 Pipeline State Object の作成
137 D3D12_COMPUTE_PIPELINE_STATE_DESC prefPsoDesc = {};
138 prefPsoDesc.pRootSignature = m_computeRootSignature.Get();
139 prefPsoDesc.CS = m_prefilterCS->GetBytecode();
140 device->CreateComputePipelineState(&prefPsoDesc, IID_PPV_ARGS(&m_prefilterPSO));
141
142 // BRDF Integration LUT用 Pipeline State Object の作成
143 D3D12_COMPUTE_PIPELINE_STATE_DESC brdfPsoDesc = {};
144 brdfPsoDesc.pRootSignature = m_computeRootSignature.Get();
145 brdfPsoDesc.CS = m_brdfCS->GetBytecode();
146 device->CreateComputePipelineState(&brdfPsoDesc, IID_PPV_ARGS(&m_brdfPSO));
147
148 SPAN_LOG("IBLBuilder Initialized Successfully.");
149 return true;
150 }

170 void IBLBuilder::GenerateCubemapFromPanorama(ID3D12Device* device, ID3D12GraphicsCommandList* cmd, D3D12_CPU_DESCRIPTOR_HANDLE panoramaSRV, Texture* outCubemap, uint32 cubemapSize)
171 {
172 if (!device || !cmd || !outCubemap) return;
173
174 // 1. 一時的なディスクリプタヒープを2枠作成
175 D3D12_DESCRIPTOR_HEAP_DESC heapDesc = {};
176 heapDesc.NumDescriptors = 2;
177 heapDesc.Type = D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV;
178 heapDesc.Flags = D3D12_DESCRIPTOR_HEAP_FLAG_SHADER_VISIBLE;
179
180 ComPtr<ID3D12DescriptorHeap> tempHeap;
181 device->CreateDescriptorHeap(&heapDesc, IID_PPV_ARGS(&tempHeap));
182 m_tempHeaps.push_back(tempHeap);
183
184 uint32 descriptorSize = device->GetDescriptorHandleIncrementSize(D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV);
185 D3D12_CPU_DESCRIPTOR_HANDLE destHandle = tempHeap->GetCPUDescriptorHandleForHeapStart();
186
187 // 2. カタログにテクスチャの情報をコピー
188 // [0] 入力パノラマ画像をコピー
189 device->CopyDescriptorsSimple(1, destHandle, panoramaSRV, D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV);
190
191 // [1] 出力Cubemapをコピー
192 destHandle.ptr += descriptorSize;
193 device->CopyDescriptorsSimple(1, destHandle, outCubemap->GetUAVCPU(), D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV);
194
195 // 3. この一時カタログをGPUにセット
196 ID3D12DescriptorHeap* heaps[] = { tempHeap.Get() };
197 cmd->SetDescriptorHeaps(1, heaps);
198
199 cmd->SetPipelineState(m_equirectToCubemapPSO.Get());
200 cmd->SetComputeRootSignature(m_computeRootSignature.Get());
201
202 // 4. スロットにカタログの住所を教える
203 D3D12_GPU_DESCRIPTOR_HANDLE gpuHandle = tempHeap->GetGPUDescriptorHandleForHeapStart();
204 cmd->SetComputeRootDescriptorTable(0, gpuHandle); // t0: パノラマ
205 gpuHandle.ptr += descriptorSize;
206 cmd->SetComputeRootDescriptorTable(1, gpuHandle); // u0: Cubemap
207
208 // 5. Compute Shader を起動
209 uint32 threadGroupsX = cubemapSize / 8;
210 uint32 threadGroupsY = cubemapSize / 8;
211 uint32 threadGroupsZ = 6;
212
213 cmd->Dispatch(threadGroupsX, threadGroupsY, threadGroupsZ);
214 }

263 void IBLBuilder::GeneratePrefilterMap(ID3D12Device* device, ID3D12GraphicsCommandList* cmd, D3D12_CPU_DESCRIPTOR_HANDLE envCubemapSRV, Texture* outPrefilterMap, uint32 baseMapSize)
264 {
265 if (!device || !cmd || !outPrefilterMap || !outPrefilterMap->GetResource()) return;
266
267 // ミップマップの最大レベルを計算
268 const uint32 maxMipLevels = 5;
269
270 // 一時ヒープを作成
271 D3D12_DESCRIPTOR_HEAP_DESC heapDesc = {};
272 heapDesc.NumDescriptors = 1 + maxMipLevels;
273 heapDesc.Type = D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV;
274 heapDesc.Flags = D3D12_DESCRIPTOR_HEAP_FLAG_SHADER_VISIBLE;
275
276 ComPtr<ID3D12DescriptorHeap> tempHeap;
277 device->CreateDescriptorHeap(&heapDesc, IID_PPV_ARGS(&tempHeap));
278 m_tempHeaps.push_back(tempHeap);
279 uint32 descSize = device->GetDescriptorHandleIncrementSize(D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV);
280
281 D3D12_CPU_DESCRIPTOR_HANDLE cpuHandle = tempHeap->GetCPUDescriptorHandleForHeapStart();
282 D3D12_GPU_DESCRIPTOR_HANDLE gpuHandle = tempHeap->GetGPUDescriptorHandleForHeapStart();
283
284 // [0] 入力環境マップ(SRV)をコピー
285 device->CopyDescriptorsSimple(1, cpuHandle, envCubemapSRV, D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV);
286
287 ID3D12DescriptorHeap* heaps[] = { tempHeap.Get() };
288 cmd->SetDescriptorHeaps(1, heaps);
289 cmd->SetPipelineState(m_prefilterPSO.Get());
290 cmd->SetComputeRootSignature(m_computeRootSignature.Get());
291 cmd->SetComputeRootDescriptorTable(0, gpuHandle); // t0: 環境Cubemap
292
293 // 各ミップレベルごとに計算をディスパッチ
294 for (uint32 mip = 0; mip < maxMipLevels; ++mip)
295 {
296 uint32 mipWidth = std::max(1u, baseMapSize >> mip);
297 float roughness = (float)mip / (float)(maxMipLevels - 1);
298
299 // 該当ミップレベル専用のUAVを作成
300 cpuHandle.ptr += descSize;
301 D3D12_UNORDERED_ACCESS_VIEW_DESC uavDesc = {};
302 uavDesc.Format = outPrefilterMap->GetResource()->GetDesc().Format;
303 uavDesc.ViewDimension = D3D12_UAV_DIMENSION_TEXTURE2DARRAY;
304 uavDesc.Texture2DArray.MipSlice = mip;
305 uavDesc.Texture2DArray.FirstArraySlice = 0;
306 uavDesc.Texture2DArray.ArraySize = 6;
307 device->CreateUnorderedAccessView(outPrefilterMap->GetResource(), nullptr, &uavDesc, cpuHandle);
308
309 // スロットにUAVとRootConstantsを設定
310 gpuHandle.ptr += descSize;
311 cmd->SetComputeRootDescriptorTable(1, gpuHandle);
312
313 // Root Constants(b0) に Roughness と Resolution を設定
314 struct ConstantData { float Roughness; float Resolution; } constants = { roughness, (float)mipWidth };
315 cmd->SetComputeRoot32BitConstants(2, 2, &constants, 0);
316
317 // ディスパッチ
318 uint32 threadGroups = std::max(1u, mipWidth / 8);
319 cmd->Dispatch(threadGroups, threadGroups, 6);
320 }
321 }
