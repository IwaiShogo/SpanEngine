50 void InspectorPanel::OnImGuiRender()
51 {
52 ImGui::Begin("Inspector", &isOpen);
53
54 // ロック機能のトグルボタン
55 float lockBtnWidth = 30.0f;
56 ImGui::SameLine(ImGui::GetWindowWidth() - lockBtnWidth - ImGui::GetStyle().WindowPadding.x);
57
58 if (m_IsLocked)
59 {
60 ImGui::PushStyleColor(ImGuiCol_Button, ImVec4(0.8f, 0.3f, 0.3f, 1.0f));
61 if (ImGui::Button(ICON_FA_LOCK, ImVec2(lockBtnWidth, 0))) m_IsLocked = false;
62 ImGui::PopStyleColor();
63 if (ImGui::IsItemHovered()) ImGui::SetTooltip("Unlock Inspector");
64 }
65 else
66 {
67 if (ImGui::Button(ICON_FA_UNLOCK, ImVec2(lockBtnWidth, 0)))
68 {
69 m_IsLocked = true;
70 m_LockedType = SelectionManager::GetType();
71 if (m_LockedType == SelectionType::Asset) m_LockedAssets = SelectionManager::GetAssetSelections();
72 if (m_LockedType == SelectionType::Entity) m_LockedEntity = SelectionManager::GetPrimaryEntity();
73 }
74 if (ImGui::IsItemHovered()) ImGui::SetTooltip("Lock Inspector");
75 }
76
77 // 1. 選択タイプの取得
78 SelectionType type = m_IsLocked ? m_LockedType : SelectionManager::GetType();
79
80 if (type == SelectionType::None)
81 {
82 // 何も選択されていない場合
83 ImGui::TextDisabled("No Selection");
84 }
85 else if (type == SelectionType::Asset)
86 {
87 // アセットインスペクターの描画
88 const auto& assets = m_IsLocked ? m_LockedAssets : SelectionManager::GetAssetSelections();
89 DrawAssetInspector(assets);
90 }
91 else if (type == SelectionType::Entity)
92 {
93 // 1. 選択ターゲットの取得
94 Entity selected = m_IsLocked ? m_LockedEntity : SelectionManager::GetPrimaryEntity();
95 World& world = Application::Get().GetWorld();
96
97 if (selected.IsNull() || !world.IsAlive(selected))
98 {
99 ImGui::TextDisabled("Invalid Entity");
100 m_IsLocked = false;
101 }
102 else
103 {
104 DrawEntityInspector(selected, world);
105 }
106 }
107
108 // ポップアップの描画
109 DrawTagEditorModal();
110 DrawLayerEditorModal();
111
112 ImGui::End();
113 }
