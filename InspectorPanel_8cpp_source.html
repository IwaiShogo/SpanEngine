55 if (selected.IsNull() || !world.IsAlive(selected))

57 ImGui::Text("No Entity Selected");
58 ImGui::End();
59 return;
60 }
61

63 ImGui::TextDisabled("ID: %d | Gen: %d", selected.ID.Index, selected.ID.Generation);
64 ImGui::Separator();
65 ImGui::Spacing();
66

70 {
71 isActive = a->IsActive;
72 }

74 {

76 }
77

79 {
80 if (Active* a = world.GetComponentPtr<Active>(selected)) a->IsActive = isActive;

91 ImGui::PushItemWidth(-1);
92 ImGui::InputText("##Name", nameComponent.Value.data(), nameComponent.Value.Capacity());
93 ImGui::PopItemWidth();
94 }
95 else
96 {
97 // Nameコンポーネントが無い場合は追加ボタンか、ダミー表示
98 char buf[64];
99 sprintf_s(buf, "Entity %d", selected.ID.Index);
100 if (ImGui::InputText("##Name", buf, sizeof(buf), ImGuiInputTextFlags_EnterReturnsTrue))
101 {
102 world.AddComponent<Name>(selected);
103 Name& n = world.GetComponent<Name>(selected);
104 n.Value = buf;
105 }
106 }
107 ImGui::PopItemWidth();
108
109 ImGui::SameLine();
110 ImGui::TextDisabled("Static");
111
112 ImGui::Spacing();
113 ImGui::Separator();

121 // [+] ボタンの幅を考慮してComboの幅を設定
122 float addBtnWidth = 30.0f;
123 ImGui::SetNextItemWidth(ImGui::GetContentRegionAvail().x * 0.5f - 10 - addBtnWidth);
124

126 const auto& registeredTags = TagManager::Get().GetAllTags();
127
128 // 現在のエンティティが持っているタグ名を取得
129 std::string currentTagStr = "Untagged";
130 if (Tag* t = world.GetComponentPtr<Tag>(selected))
131 {
132 currentTagStr = t->Value.data();
133 }
134 else
135 {
136 world.AddComponent<Tag>(selected, Tag("Untagged"));
137 }
138
139 // 現在のタグがリストの何番目か
140 int currentTagIndex = 0;
141 for (int i = 0; i < registeredTags.size(); i++)
142 {
143 if (registeredTags[i] == currentTagStr)
144 {
145 currentTagIndex = i;
146 break;
147 }
148 }
149
150 // ImGui::Combo用のC文字列配列を作成
151 std::vector<const char*> tagCStrs;
152 tagCStrs.reserve(registeredTags.size());
153 for (const auto& tag : registeredTags) tagCStrs.push_back(tag.c_str());
154
155 if (ImGui::Combo("##Tag", &currentTagIndex, tagCStrs.data(), (int)tagCStrs.size()))
156 {
157 // 変更をコンポーネントに適用
158 if (Tag* t = world.GetComponentPtr<Tag>(selected))
159 {
160 t->Value = registeredTags[currentTagIndex];
161 }
162 }
163
164 // [+] タグ追加ボタン
165 ImGui::SameLine();
166 if (ImGui::Button("+##AddTag", ImVec2(addBtnWidth, 0)))
167 {
168 ImGui::OpenPopup("AddTagPopup");

172 if (ImGui::BeginPopup("AddTagPopup"))

175 ImGui::Separator();

180 std::string newTag(newTagBuffer);
181 if (TagManager::Get().AddTag(newTag))
182 {
183 // 追加後、即座にそのタグを設定する
184 if (Tag* t = world.GetComponentPtr<Tag>(selected))
185 {
186 t->Value = newTag;
187 }
188 }
189 newTagBuffer[0] = '\n';
190 ImGui::CloseCurrentPopup();
191 }
192 ImGui::EndPopup();
193 }
194
195 ImGui::SameLine();
196
197 // --- Layer ---
198 ImGui::Text("Layer"); ImGui::SameLine(labelWidth + ImGui::GetContentRegionAvail().x * 0.5f + 5);
199 ImGui::SetNextItemWidth(ImGui::GetContentRegionAvail().x);
200
201 const char* layers[] = { "Default", "Transparent", "Ignore Raycast", "Water", "UI" };
202 int currentLayerIndex = 0;
203
204 if (Layer* l = world.GetComponentPtr<Layer>(selected))
205 {
206 currentLayerIndex = l->Value;
207 if (currentLayerIndex < 0 || currentLayerIndex >= IM_ARRAYSIZE(layers)) currentLayerIndex = 0;
208 }
209 else
210 {
211 world.AddComponent<Layer>(selected, Layer(0));
212 }
213
214 if (ImGui::Combo("##Layer", &currentLayerIndex, layers, IM_ARRAYSIZE(layers)))
215 {
216 if (Layer* l = world.GetComponentPtr<Layer>(selected)) {
217 l->Value = currentLayerIndex;
218 }
219 }
220 }
221
222 ImGui::Spacing();
223 ImGui::Separator();
224 ImGui::Spacing();
225
226 // 4. コンポーネントリストの描画 (Reflection)
227 // 登録されている全コンポーネントメタデータを取得
228 auto& components = ComponentRegistry::GetAll();
229
230 // ソート (Transformを最優先、あとは名前順)
231 std::sort(components.begin(), components.end(), [](const ComponentMetadata& a, const ComponentMetadata& b)
232 {
233 if (a.Name == "Transform") return true;
234 if (b.Name == "Transform") return false;
235
236 return a.Order < b.Order;
237 });
238
239 // 描画ループ
240 for (size_t i = 0; i < components.size(); ++i)
241 {
242 auto& meta = components[i];
243
244 bool isRemove = false;
245
246 // 基本コンポーネントはリストに出さない
247 if (meta.Name == "Name" || meta.Name == "Tag" || meta.Name == "Layer" || meta.Name == "Active") continue;
248
249 ImGui::PushID(i);
250
251 // 描画実行
252 meta.DrawFunc(selected, world);
253
254 ImGui::PopID();
255 }
256
257 // 5. コンポーネント追加ボタン (Add Component)
258 ImGui::Spacing();
259 ImGui::Separator();
260
261 if (ImGui::Button("Add Component", ImVec2(-1, 0)))
262 {
263 ImGui::OpenPopup("AddComponentPopup");
264 }
265
266 // 追加ポップアップ
267 if (ImGui::BeginPopup("AddComponentPopup"))
268 {
269 // コンポーネント名をソートして表示
270 std::vector<const ComponentMetadata*> sortedMetas;
271 for (const auto& meta : components) sortedMetas.push_back(&meta);
272 std::sort(sortedMetas.begin(), sortedMetas.end(), [](const auto* a, const auto* b) { return a->Name < b->Name; });
273
274 for (const auto* meta : sortedMetas)
275 {
276 if (meta->Name == "Name" || meta->Name == "Tag" || meta->Name == "Layer" || meta->Name == "Active") continue;
277
278 if (meta->HasFunc && meta->HasFunc(selected, world)) continue;
279
280 if (ImGui::MenuItem(meta->Name.c_str()))
281 {
282 if (meta->Name.c_str())
283 {
284 meta->AddFunc(selected, world);
285 }
286 }
287 }
288 ImGui::EndPopup();
289 }
290 }

292 ImGui::End();
