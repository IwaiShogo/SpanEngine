43 void InspectorPanel::OnImGuiRender()
44 {
45 ImGui::Begin("Inspector", &isOpen);
46
47 // 1. 選択タイプの取得
48 SelectionType type = SelectionManager::GetType();
49
50 if (type == SelectionType::None)
51 {
52 // 何も選択されていない場合
53 ImGui::TextDisabled("No Selection");
54 }
55 else if (type == SelectionType::Asset)
56 {
57 // アセットインスペクターの描画
58 const auto& assets = SelectionManager::GetAssetSelections();
59 DrawAssetInspector(assets);
60 }
61 else if (type == SelectionType::Entity)
62 {
63 // 1. 選択ターゲットの取得
64 Entity selected = SelectionManager::GetPrimaryEntity();
65 World& world = Application::Get().GetWorld();
66
67 if (selected.IsNull() || !world.IsAlive(selected))
68 {
69 ImGui::TextDisabled("Invalid Entity");
70 }
71 else
72 {
73 DrawEntityInspector(selected, world);
74 }
75 }
76
77 // ポップアップの描画
78 DrawTagEditorModal();
79 DrawLayerEditorModal();
80
81 ImGui::End();
82 }

362 // 4. コンポーネントリストの描画

365 // ソート (Transformを最優先、あとは名前順)
366 std::sort(components.begin(), components.end(), [](const ComponentMetadata& a, const ComponentMetadata& b)
367 {
368 if (a.Name == "Transform") return true;
369 if (b.Name == "Transform") return false;
370
371 return a.Order < b.Order;
372 });
373
374 // 描画ループ
375 for (size_t i = 0; i < components.size(); ++i)
376 {
377 auto& meta = components[i];
378
379 // 基本コンポーネントはリストに出さない

391 ImGui::Spacing();
392 ImGui::Separator();
393
394 if (ImGui::Button("Add Component", ImVec2(-1, 0)))
395 {
396 ImGui::OpenPopup("AddComponentPopup");
397 }
398
399 // 追加ポップアップ
400 if (ImGui::BeginPopup("AddComponentPopup"))
401 {
402 // コンポーネント名をソートして表示
403 std::vector<const ComponentMetadata*> sortedMetas;
404 for (const auto& meta : components) sortedMetas.push_back(&meta);
405 std::sort(sortedMetas.begin(), sortedMetas.end(), [](const auto* a, const auto* b) { return a->Name < b->Name; });
406
407 for (const auto* meta : sortedMetas)
408 {
409 if (meta->Name == "Name" || meta->Name == "Tag" || meta->Name == "Layer" || meta->Name == "Active") continue;
410
411 if (meta->HasFunc && meta->HasFunc(selected, world)) continue;
412
413 if (ImGui::MenuItem(meta->Name.c_str()))
414 {
415 if (meta->Name.c_str())
416 {
417 meta->AddFunc(selected, world);
418 }
419 }
420 }
421 ImGui::EndPopup();
