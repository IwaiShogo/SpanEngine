102 void SkyboxPass::Render(Renderer* renderer, ID3D12GraphicsCommandList* cmd, const EnvironmentSettings& env, const Matrix4x4& view, const Matrix4x4& proj, const Vector3& camPos, Texture* envCubemap)
103 {
104 if (!m_pso) return;
105
106 struct SkyboxCameraCB { Matrix4x4 InvView; Matrix4x4 InvProj; Vector3 CamPos; float pad; };
107 struct SkyboxSettingsCB
108 {
109 Vector3 TopColor;
110 float pad1;
111 Vector3 HorizonColor;
112 float pad2;
113 Vector3 BottomColor;
114 int SkyMode;
115 float Exposure;
116 float pad3[3];
117 };
118
119 SkyboxCameraCB camData;
120 camData.InvView.FromXM(XMMatrixTranspose(XMMatrixInverse(nullptr, view.ToXM())));
121 camData.InvProj.FromXM(XMMatrixTranspose(XMMatrixInverse(nullptr, proj.ToXM())));
122 camData.CamPos = camPos;
123
124 SkyboxSettingsCB setData;
125 setData.TopColor = env.SkyTopColor;
126 setData.HorizonColor = env.SkyHorizonColor;
127 setData.BottomColor = env.SkyTopColor;
128
129 setData.SkyMode = (envCubemap != nullptr) ? 1 : 0;
130 setData.Exposure = env.Exposure;
131
132 // Renderer„ÅÆÊ©üËÉΩ„Çí‰Ωø„Å£„Å¶CBV„ÇíÂâ≤„ÇäÂΩì„Å¶„Å¶GPU„Ç¢„Éâ„É¨„Çπ„ÇíÂèñÂæó

134 D3D12_GPU_VIRTUAL_ADDRESS cbSetAddress = renderer->AllocateCBV(&setData, sizeof(SkyboxSettingsCB));
135
136 if (cbCamAddress == 0 || cbSetAddress == 0) return;

138 cmd->SetPipelineState(m_pso.Get());
