86 m_currentLightData.ScreenHeight = screenHeight;
87

117 void LightManager::ExecuteLightCulling(Renderer* renderer, ID3D12GraphicsCommandList* cmd, const Matrix4x4& viewMatrix, const Matrix4x4& projectionMatrix, uint32 screenWidth, uint32 screenHeight, RenderTarget* gBuffer)
118 {
119 if (!renderer || !cmd || !m_frustumsBuffer || !gBuffer || screenWidth == 0 || screenHeight == 0) return;
120
121 // üåü „Éí„Éº„Éó„ÅÆ‰∏äÊõ∏„Åç„Çí„ÇÑ„ÇÅ„ÄÅRenderer „Å´„Éê„Ç§„É≥„Éâ„Çí‰ªª„Åõ„Çã
122 cmd->SetComputeRootSignature(m_computeRootSignature.Get());
123
124 uint32 numTilesX = (screenWidth + TILE_SIZE - 1) / TILE_SIZE;
125 uint32 numTilesY = (screenHeight + TILE_SIZE - 1) / TILE_SIZE;
126
127 // ÂÆöÊï∞„ÅÆ„Çª„ÉÉ„Éà (InvProjection, View, Dimensions)
128 DirectX::XMVECTOR det; DirectX::XMMATRIX invProj = DirectX::XMMatrixInverse(&det, projectionMatrix.ToXM());
129 Matrix4x4 invProjMat; invProjMat.FromXM(XMMatrixTranspose(invProj));
130 Matrix4x4 viewMatTransposed; viewMatTransposed.FromXM(XMMatrixTranspose(viewMatrix.ToXM()));
131
132 struct CullingCB { Matrix4x4 InvProj; Matrix4x4 View; uint32 Dim[2]; uint32 Tiles[2]; } cbData;
133 cbData.InvProj = invProjMat; cbData.View = viewMatTransposed;
134 cbData.Dim[0] = screenWidth; cbData.Dim[1] = screenHeight; cbData.Tiles[0] = numTilesX; cbData.Tiles[1] = numTilesY;
135
136 cmd->SetComputeRoot32BitConstants(0, sizeof(CullingCB) / 4, &cbData, 0);
137 cmd->SetComputeRootConstantBufferView(1, m_lightConstantBuffer->GetGPUVirtualAddress());
138
139 // „Éê„ÉÉ„Éï„Ç°Áæ§„ÇíÈ†Ü„Å´„Éê„Ç§„É≥„Éâ [2]~[7]
140 renderer->BindComputeSRV(cmd, m_lightDataBuffer->GetSRV(), 2); // t0
141 renderer->BindComputeSRV(cmd, gBuffer->GetSRV(), 3); // t1 (GBuffer Depth)
142 renderer->BindComputeUAV(cmd, m_frustumsBuffer->GetUAV(), 4); // u0
143 renderer->BindComputeUAV(cmd, m_lightIndexCounter->GetUAV(), 5); // u1
144 renderer->BindComputeUAV(cmd, m_lightIndexList->GetUAV(), 6); // u2
145 renderer->BindComputeUAV(cmd, m_lightGrid->GetUAV(), 7); // u3
146
147 uint32 dispatchX = (numTilesX + TILE_SIZE - 1) / TILE_SIZE;
148 uint32 dispatchY = (numTilesY + TILE_SIZE - 1) / TILE_SIZE;
149
150 // ÂÆüË°å
151 cmd->SetPipelineState(m_psoFrustums.Get()); cmd->Dispatch(dispatchX, dispatchY, 1);
152 cmd->SetPipelineState(m_psoResetCounter.Get()); cmd->Dispatch(1, 1, 1);
153
154 D3D12_RESOURCE_BARRIER barriers[2] = {};
155 barriers[0].Type = D3D12_RESOURCE_BARRIER_TYPE_UAV; barriers[0].UAV.pResource = m_frustumsBuffer->GetResource();
156 barriers[1].Type = D3D12_RESOURCE_BARRIER_TYPE_UAV; barriers[1].UAV.pResource = m_lightIndexCounter->GetResource();
157 cmd->ResourceBarrier(2, barriers);
158
159 cmd->SetPipelineState(m_psoCulling.Get()); cmd->Dispatch(numTilesX, numTilesY, 1);
160 }

214 psoDesc.CS = m_shaderFrustums->GetBytecode();
