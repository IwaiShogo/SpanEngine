37 void LightManager::OnResize(ID3D12Device* device, uint32 width, uint32 height)
38 {
39 if (!device || width == 0 || height == 0) return;
40
41 // タイル数の計算
42 uint32 numTilesX = (width + TILE_SIZE - 1) / TILE_SIZE;
43 uint32 numTilesY = (height + TILE_SIZE - 1) / TILE_SIZE;
44 uint32 numTiles = numTilesX * numTilesY;
45
46 // バッファを解放して再作成
47 m_lightIndexCounter.reset();
48 m_lightIndexList.reset();
49 m_lightGrid.reset();
50
51 // 1. Opaque Light Index Counter
52 m_lightIndexCounter = std::make_unique<ComputeBuffer>();
53 m_lightIndexCounter->Initialize(device, sizeof(uint32), 1, true);
54
55 // 2. Opaque Light Index List
56 m_lightIndexList = std::make_unique<ComputeBuffer>();
57 m_lightIndexList->Initialize(device, sizeof(uint32), numTiles * MAX_LIGHTS_PER_TILE, true);
58
59 // 3. Light Grid
60 m_lightGrid = std::make_unique<ComputeBuffer>();
61 m_lightGrid->Initialize(device, sizeof(uint32) * 2, numTiles, true);
62
63 // 4. Frustums Buffer
64 m_frustumsBuffer = std::make_unique<ComputeBuffer>();
65 m_frustumsBuffer->Initialize(device, sizeof(float) * 16, numTiles, true);
66
67 // ComputeShader用のDescriptorHeapを作成し、バッファ群をコピー
68 D3D12_DESCRIPTOR_HEAP_DESC heapDesc = {};
69 heapDesc.NumDescriptors = 5;
70 heapDesc.Type = D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV;
71 heapDesc.Flags = D3D12_DESCRIPTOR_HEAP_FLAG_SHADER_VISIBLE;
72 device->CreateDescriptorHeap(&heapDesc, IID_PPV_ARGS(&m_computeDescriptorHeap));
73
74 uint32 increment = device->GetDescriptorHandleIncrementSize(D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV);
75 D3D12_CPU_DESCRIPTOR_HANDLE destCpu = m_computeDescriptorHeap->GetCPUDescriptorHandleForHeapStart();
76
77 device->CopyDescriptorsSimple(1, destCpu, m_lightDataBuffer->GetSRV(), D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV); destCpu.ptr += increment;
78 device->CopyDescriptorsSimple(1, destCpu, m_frustumsBuffer->GetUAV(), D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV); destCpu.ptr += increment;
79 device->CopyDescriptorsSimple(1, destCpu, m_lightIndexCounter->GetUAV(), D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV); destCpu.ptr += increment;
80 device->CopyDescriptorsSimple(1, destCpu, m_lightIndexList->GetUAV(), D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV); destCpu.ptr += increment;
81 device->CopyDescriptorsSimple(1, destCpu, m_lightGrid->GetUAV(), D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV);
82
83 SPAN_LOG("LightManager Resized: %dx%d (%d Tiles)", width, height, numTiles);
84 }

86 void LightManager::UpdateLightData(const std::vector<LightDataGPU>& lights, const EnvironmentSettings& env, const Vector3& cameraPos, bool enableSSAO, uint32 screenWidth, uint32 screenHeight)
87 {
88 m_currentLightData.CameraPosition = cameraPos;
89 m_currentLightData.Exposure = env.Exposure;
90 m_currentLightData.AmbientIntensity = env.AmbientIntensity;
91 m_currentLightData.EnvReflectionIntensity = env.EnvReflectionIntensity;
92
93 // IBLカラーをLiner
94 m_currentLightData.SkyTopColor = Vector3(pow(env.SkyTopColor.x, 2.2f), pow(env.SkyTopColor.y, 2.2f), pow(env.SkyTopColor.z, 2.2f));
95 m_currentLightData.SkyHorizonColor = Vector3(pow(env.SkyHorizonColor.x, 2.2f), pow(env.SkyHorizonColor.y, 2.2f), pow(env.SkyHorizonColor.z, 2.2f));
96 m_currentLightData.SkyBottomColor = Vector3(pow(env.SkyBottomColor.x, 2.2f), pow(env.SkyBottomColor.y, 2.2f), pow(env.SkyBottomColor.z, 2.2f));
97
98 m_currentLightData.SkyMode = (env.Mode == SkyboxMode::HDRI) ? 1 : 0;
99 m_currentLightData.EnableSSAO = enableSSAO ? 1 : 0;
100 m_currentLightData.ScreenWidth = screenWidth;
101 m_currentLightData.ScreenHeight = screenHeight;
102
103 m_currentLightData.DirectionalLightSpaceMatrix = Matrix4x4::Identity().Transpose();
104 m_currentLightData.ActiveLightCount = std::min((int)lights.size(), MAX_LIGHTS);
105
106 // CPU上で配列を作って一気に転送
107 std::vector<LightDataGPU> lightDataArray;
108 lightDataArray.reserve(m_currentLightData.ActiveLightCount);
109
110 for (int i = 0; i < m_currentLightData.ActiveLightCount; ++i)
111 {
112 lightDataArray.push_back(lights[i]);
113 if (lights[i].Type == 0) m_currentLightData.DirectionalLightSpaceMatrix = lights[i].ShadowMatrix.Transpose();
114 }
115
116 if (m_lightConstantBuffer)
117 {
118 m_lightConstantBuffer->Update(m_currentLightData);
119 }
120
121 if (m_lightDataBuffer && !lightDataArray.empty())
122 {
123 m_lightDataBuffer->UpdateData(lightDataArray.data(), lightDataArray.size() * sizeof(LightDataGPU));
124 }
125 }

132 void LightManager::ExecuteLightCulling(ID3D12Device* device, ID3D12GraphicsCommandList* cmd, const Matrix4x4& viewMatrix, const Matrix4x4& projectionMatrix, uint32 screenWidth, uint32 screenHeight)
133 {
134 if (!device || !cmd || !m_frustumsBuffer || !m_computeDescriptorHeap || screenWidth == 0 || screenHeight == 0) return;
135
136 // 定数の準備とセット
137 uint32 numTilesX = (screenWidth + TILE_SIZE - 1) / TILE_SIZE;
138 uint32 numTilesY = (screenHeight + TILE_SIZE - 1) / TILE_SIZE;
139
140 DirectX::XMVECTOR det;
141 DirectX::XMMATRIX invProj = DirectX::XMMatrixInverse(&det, projectionMatrix.ToXM());
142 Matrix4x4 invProjMat;
143 invProjMat.FromXM(XMMatrixTranspose(invProj));
144
145 Matrix4x4 viewMatTransposed;
146 viewMatTransposed.FromXM(XMMatrixTranspose(viewMatrix.ToXM()));
147
148 struct CullingCB
149 {
150 Matrix4x4 InvProjection;
151 Matrix4x4 View;
152 uint32 ScreenDim[2];
153 uint32 NumTiles[2];
154 } cbData;
155 cbData.InvProjection = invProjMat;
156 cbData.View = viewMatTransposed;
157 cbData.ScreenDim[0] = screenWidth;
158 cbData.ScreenDim[1] = screenHeight;
159 cbData.NumTiles[0] = numTilesX;
160 cbData.NumTiles[1] = numTilesY;
161
162 ID3D12DescriptorHeap* heaps[] = { m_computeDescriptorHeap.Get() };
163 cmd->SetDescriptorHeaps(1, heaps);
164
165 // [0] Constants, [1] LightBufferCBV, [2] SRV Table(1個), [3] UAV Table(4個)
166 cmd->SetComputeRootSignature(m_computeRootSignature.Get());
167 cmd->SetComputeRoot32BitConstants(0, sizeof(CullingCB) / 4, &cbData, 0);
168 cmd->SetComputeRootConstantBufferView(1, m_lightConstantBuffer->GetGPUVirtualAddress());
169 cmd->SetComputeRootDescriptorTable(2, m_computeDescriptorHeap->GetGPUDescriptorHandleForHeapStart());
170
171 D3D12_GPU_DESCRIPTOR_HANDLE uavStart = m_computeDescriptorHeap->GetGPUDescriptorHandleForHeapStart();
172 uavStart.ptr += device->GetDescriptorHandleIncrementSize(D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV); // 1つ飛ばす(SRVの次)
173 cmd->SetComputeRootDescriptorTable(3, uavStart);
174
175 uint32 dispatchX = (numTilesX + TILE_SIZE - 1) / TILE_SIZE;
176 uint32 dispatchY = (numTilesY + TILE_SIZE - 1) / TILE_SIZE;
177
178 // 1. フラスタム計算パス
179 cmd->SetPipelineState(m_psoFrustums.Get());
180 cmd->Dispatch(dispatchX, dispatchY, 1);
181
182 // 2. カウンタのリセットパス
183 cmd->SetPipelineState(m_psoResetCounter.Get());
184 cmd->Dispatch(1, 1, 1);
185
186 // UAVバリア (リセットとフラスタム計算の完了を待つ)
187 D3D12_RESOURCE_BARRIER barriers[2] = {};
188 barriers[0].Type = D3D12_RESOURCE_BARRIER_TYPE_UAV; barriers[0].UAV.pResource = m_frustumsBuffer->GetResource();
189 barriers[1].Type = D3D12_RESOURCE_BARRIER_TYPE_UAV; barriers[1].UAV.pResource = m_lightIndexCounter->GetResource();
190 cmd->ResourceBarrier(2, barriers);
191
192 // 3. ライトカリングパス
193 cmd->SetPipelineState(m_psoCulling.Get());
194 cmd->Dispatch(numTilesX, numTilesY, 1);
195 }

197 bool LightManager::InitializeCompute(ID3D12Device* device)
198 {
199 // 1. シェーダーのロード
200 m_shaderFrustums = std::make_unique<Shader>();
201 if (!m_shaderFrustums->Load(L"LightCulling.hlsl", ShaderType::Compute, "CS_ComputeFrustums")) return false;
202
203 // カリングシェーダーのロード
204 m_shaderCulling = std::make_unique<Shader>();
205 if (!m_shaderCulling->Load(L"LightCulling.hlsl", ShaderType::Compute, "CS_LightCulling")) return false;
206
207 // 2. Root Signature の作成
208 D3D12_ROOT_PARAMETER rootParams[4];
209
210 // b0: CullingCB (32bit Constants)
211 rootParams[0].ParameterType = D3D12_ROOT_PARAMETER_TYPE_32BIT_CONSTANTS;
212 rootParams[0].Constants.ShaderRegister = 0;
213 rootParams[0].Constants.RegisterSpace = 0;
214 rootParams[0].Constants.Num32BitValues = 36;
215 rootParams[0].ShaderVisibility = D3D12_SHADER_VISIBILITY_ALL;
216
217 // b1: LightBuffer (CBV)
218 rootParams[1].ParameterType = D3D12_ROOT_PARAMETER_TYPE_CBV;
219 rootParams[1].Descriptor.ShaderRegister = 1;
220 rootParams[1].Descriptor.RegisterSpace = 0;
221 rootParams[1].ShaderVisibility = D3D12_SHADER_VISIBILITY_ALL;
222
223 // t0: LightData Array (SRV)
224 D3D12_DESCRIPTOR_RANGE srvRange = {};
225 srvRange.RangeType = D3D12_DESCRIPTOR_RANGE_TYPE_SRV;
226 srvRange.NumDescriptors = 1;
227 srvRange.BaseShaderRegister = 0;
228 srvRange.OffsetInDescriptorsFromTableStart = 0;
229
230 rootParams[2].ParameterType = D3D12_ROOT_PARAMETER_TYPE_DESCRIPTOR_TABLE;
231 rootParams[2].DescriptorTable.NumDescriptorRanges = 1;
232 rootParams[2].DescriptorTable.pDescriptorRanges = &srvRange;
233 rootParams[2].ShaderVisibility = D3D12_SHADER_VISIBILITY_ALL;
234
235 // u0 ~ u3: UAV Table
236 D3D12_DESCRIPTOR_RANGE uavRange = {};
237 uavRange.RangeType = D3D12_DESCRIPTOR_RANGE_TYPE_UAV;
238 uavRange.NumDescriptors = 4;
239 uavRange.BaseShaderRegister = 0;
240 uavRange.OffsetInDescriptorsFromTableStart = 0;
241
242 rootParams[3].ParameterType = D3D12_ROOT_PARAMETER_TYPE_DESCRIPTOR_TABLE;
243 rootParams[3].DescriptorTable.NumDescriptorRanges = 1;
244 rootParams[3].DescriptorTable.pDescriptorRanges = &uavRange;
245 rootParams[3].ShaderVisibility = D3D12_SHADER_VISIBILITY_ALL;
246
247 D3D12_ROOT_SIGNATURE_DESC rootSigDesc = {};
248 rootSigDesc.NumParameters = 4;
249 rootSigDesc.pParameters = rootParams;
250
251 ComPtr<ID3DBlob> signature, error;
252 if (FAILED(D3D12SerializeRootSignature(&rootSigDesc, D3D_ROOT_SIGNATURE_VERSION_1, &signature, &error))) return false;
253 if (FAILED(device->CreateRootSignature(0, signature->GetBufferPointer(), signature->GetBufferSize(), IID_PPV_ARGS(&m_computeRootSignature)))) return false;
254
255 // 3. Pipeline State の作成
256 D3D12_COMPUTE_PIPELINE_STATE_DESC psoDesc = {};
257 psoDesc.pRootSignature = m_computeRootSignature.Get();
258 psoDesc.CS = m_shaderFrustums->GetBytecode();
259 if (FAILED(device->CreateComputePipelineState(&psoDesc, IID_PPV_ARGS(&m_psoFrustums)))) return false;
260
261 m_shaderResetCounter = std::make_unique<Shader>();
262 if (!m_shaderResetCounter->Load(L"LightCulling.hlsl", ShaderType::Compute, "CS_ResetCounter")) return false;
263
264 // カリング用PSOの作成
265 psoDesc.CS = m_shaderCulling->GetBytecode();
266 if (FAILED(device->CreateComputePipelineState(&psoDesc, IID_PPV_ARGS(&m_psoCulling)))) return false;
267
268 return true;
269 }
